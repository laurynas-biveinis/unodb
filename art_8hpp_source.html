342 void push(detail::node_ptr node, std::byte key_byte,
343 std::uint8_t child_index, detail::key_prefix_snapshot prefix) {
344 // For variable length keys we need to know the number of bytes
345 // associated with the node's key_prefix. In addition there is
346 // one byte for the descent to the child node along the
347 // child_index. That information needs to be stored on the
348 // stack so we can pop off the right number of bytes even for
349 // OLC where the node might be concurrently modified.
350 UNODB_DETAIL_ASSERT(node.type() != node_type::LEAF);
351 stack_.push({node, key_byte, child_index, prefix});
352 keybuf_.push(prefix.get_key_view());
353 keybuf_.push(key_byte);
354 }
355
357 void push_leaf(detail::node_ptr aleaf) {
358 // Mock up a stack entry for the leaf.
359 stack_.push({
360 aleaf,
361 static_cast<std::byte>(0xFFU), // ignored for leaf
362 static_cast<std::uint8_t>(0xFFU), // ignored for leaf
363 detail::key_prefix_snapshot(0) // ignored for leaf
364 });
365 // No change in the key_buffer.
366 }
367
369 void push(const typename inode_base::iter_result& e) {
370 const auto node_type = e.node.type();
371 if (UNODB_DETAIL_UNLIKELY(node_type == node_type::LEAF)) {
372 push_leaf(e.node);
373 }
374 push(e.node, e.key_byte, e.child_index, e.prefix);
375 }

379 UNODB_DETAIL_ASSERT(!empty());

381 const auto prefix_len = top().prefix.length();
382 keybuf_.pop(prefix_len);
383 stack_.pop();
384 }

806inline auto* unwrap_fake_critical_section(

812template <typename Key, typename Value, class INode>
813detail::node_ptr* impl_helpers::add_or_choose_subtree(
814 INode& inode, std::byte key_byte, basic_art_key<Key> k, value_view v,
815 db<Key, Value>& db_instance, tree_depth<basic_art_key<Key>> depth,
816 detail::node_ptr* node_in_parent) {
817 auto* const child =
818 unwrap_fake_critical_section(inode.find_child(key_byte).second);
819
820 if (child != nullptr) return child;
821
822 auto leaf = art_policy<Key, Value>::make_db_leaf_ptr(k, v, db_instance);
823 const auto children_count = inode.get_children_count();
824
825 if constexpr (!std::is_same_v<INode, inode_256<Key, Value>>) {
826 if (UNODB_DETAIL_UNLIKELY(children_count == INode::capacity)) {
827 auto larger_node{INode::larger_derived_type::create(
828 db_instance, inode, std::move(leaf), depth)};
829 *node_in_parent =
830 node_ptr{larger_node.release(), INode::larger_derived_type::type};
831#ifdef UNODB_DETAIL_WITH_STATS
832 db_instance
833 .template account_growing_inode<INode::larger_derived_type::type>();
834#endif // UNODB_DETAIL_WITH_STATS
835 return child;
836 }
837 }
838 inode.add_to_nonfull(std::move(leaf), depth, children_count);
839 return child;
840}

842// MSVC C26815 false positive: create() returns smart pointer with LIFETIMEBOUND
843// on db param, but release() transfers ownership and the raw pointer's validity
844// is independent of the temporary unique_ptr.
845UNODB_DETAIL_DISABLE_MSVC_WARNING(26815)
846template <typename Key, typename Value, class INode>
847std::optional<detail::node_ptr*> impl_helpers::remove_or_choose_subtree(
848 INode& inode, std::byte key_byte, basic_art_key<Key> k,
849 db<Key, Value>& db_instance, detail::node_ptr* node_in_parent) {
850 const auto [child_i, child_ptr]{inode.find_child(key_byte)};
851
852 if (child_ptr == nullptr) return {};
853
854 const auto child_ptr_val{child_ptr->load()};
855 if (child_ptr_val.type() != node_type::LEAF)
856 return unwrap_fake_critical_section(child_ptr);
857
858 const auto* const leaf{
859 child_ptr_val.template ptr<typename db<Key, Value>::leaf_type*>()};
860 if (!leaf->matches(k)) return {};
861
862 if (UNODB_DETAIL_UNLIKELY(inode.is_min_size())) {
863 if constexpr (std::is_same_v<INode, inode_4<Key, Value>>) {

929template <typename Key, typename Value>
930bool db<Key, Value>::insert_internal(art_key_type insert_key, value_type v) {
931 if (UNODB_DETAIL_UNLIKELY(root == nullptr)) {
932 auto leaf = art_policy::make_db_leaf_ptr(insert_key, v, *this);
933 root = detail::node_ptr{leaf.release(), node_type::LEAF};
934 return true;
935 }
936
937 auto* node = &root;
938 tree_depth_type depth{};
939 auto remaining_key{insert_key};
940
941 while (true) {
942 const auto node_type = node->type();
943 if (node_type == node_type::LEAF) {
944 auto* const leaf{node->template ptr<leaf_type*>()};
945 const auto existing_key{leaf->get_key_view()};
946 const auto cmp = insert_key.cmp(existing_key);
947 if (UNODB_DETAIL_UNLIKELY(cmp == 0)) {
948 return false; // exists
949 }
950 // Replace the existing leaf with a new N4 and put the existing
951 // leaf and the leaf for the caller's key and value under the
952 // new inode as its direct children.
953 auto new_leaf = art_policy::make_db_leaf_ptr(insert_key, v, *this);
954 auto new_node{inode_4::create(*this, existing_key, remaining_key, depth,
955 leaf, std::move(new_leaf))};
956 *node = detail::node_ptr{new_node.release(), node_type::I4};
957#ifdef UNODB_DETAIL_WITH_STATS
958 account_growing_inode<node_type::I4>();
959#endif // UNODB_DETAIL_WITH_STATS
960 return true;
961 }
962
963 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
964
965 auto* const inode{node->template ptr<inode_type*>()};
966 const auto& key_prefix{inode->get_key_prefix()};
967 const auto key_prefix_length{key_prefix.length()};
968 const auto shared_prefix_len{key_prefix.get_shared_length(remaining_key)};
969 if (shared_prefix_len < key_prefix_length) {
970 // We have reached an existing inode whose key_prefix is greater
971 // than the desired match. We need to split this inode into a
972 // new N4 whose children are the existing inode and a new child
973 // leaf.
974 auto leaf = art_policy::make_db_leaf_ptr(insert_key, v, *this);
975 auto new_node = inode_4::create(*this, *node, shared_prefix_len, depth,
976 std::move(leaf));
977 *node = detail::node_ptr{new_node.release(), node_type::I4};
978#ifdef UNODB_DETAIL_WITH_STATS
979 account_growing_inode<node_type::I4>();
980 ++key_prefix_splits;
981 UNODB_DETAIL_ASSERT(growing_inode_counts[internal_as_i<node_type::I4>] >
982 key_prefix_splits);
983#endif // UNODB_DETAIL_WITH_STATS
984 return true;
985 }
986 // key_prefix bytes were absorbed during the descent. Now we need
987 // to either descend along an existing child or create a new child.
988 UNODB_DETAIL_ASSERT(shared_prefix_len == key_prefix_length);
989 depth += key_prefix_length;
990 remaining_key.shift_right(key_prefix_length);
991
992 node = inode->template add_or_choose_subtree<detail::node_ptr*>(
993 node_type, remaining_key[0], insert_key, v, *this, depth, node);
994
995 if (node == nullptr) return true;
996
997 ++depth;
998 remaining_key.shift_right(1);
999 }

1079// Position the iterator on the next leaf in the index.
1080template <typename Key, typename Value>
1081typename db<Key, Value>::iterator& db<Key, Value>::iterator::next() {
1082 while (!empty()) {
1083 const auto& e = top();
1084 const auto node{e.node};
1085 UNODB_DETAIL_ASSERT(node != nullptr);
1086 const auto node_type = node.type();
1087 if (node_type == node_type::LEAF) {
1088 pop(); // pop off the leaf
1089 continue; // falls through loop if just a root leaf since stack now
1090 // empty.
1091 }
1092 auto* inode{node.template ptr<inode_type*>()};
1093 const auto nxt = inode->next(node_type,
1094 e.child_index); // next child of that parent.
1095 if (!nxt.has_value()) {
1096 pop(); // Nothing more for that inode.
1097 continue; // We will look for the right sibling of the parent inode.
1098 }
1099 // Fix up stack for new parent node state and left-most descent.
1100 const auto& e2 = nxt.value();
1101 pop();
1102 push(e2);
1103 const auto child = inode->get_child(node_type, e2.child_index); // descend
1104 return left_most_traversal(child);
1105 }
1106 return *this; // stack is empty, so iterator == end().
1107}

1114 const auto node{e.node};

1116 const auto node_type = node.type();

1118 pop(); // pop off the leaf

1126 continue; // We will look for the left sibling of the parent inode.

1132 push(e2);
1133 auto child = inode->get_child(node_type, e2.child_index); // descend
1134 return right_most_traversal(child);

1140// node to the left-most leaf under that node, pushing nodes onto the
1141// stack as they are visited.
1142template <typename Key, typename Value>
1143typename db<Key, Value>::iterator&
1144db<Key, Value>::iterator::left_most_traversal(detail::node_ptr node) {
1145 while (true) {
1146 UNODB_DETAIL_ASSERT(node != nullptr);
1147 const auto node_type = node.type();

1149 push_leaf(node);
1150 return *this; // done
1151 }
1152 // recursive descent.
1153 auto* const inode{node.ptr<inode_type*>()};
1154 const auto e =
1155 inode->begin(node_type); // first child of current internal node
1156 push(e); // push the entry on the stack.
1157 node = inode->get_child(node_type, e.child_index); // get the child
1158 }
1159 UNODB_DETAIL_CANNOT_HAPPEN();

1185// Note: The basic seek() logic is similar to ::get() as long as the
1186// search_key exists in the data. However, the iterator is positioned
1187// instead of returning the value for the key. Life gets a lot more
1188// complicated when the search_key is not in the data and we have to
1189// consider the cases for both forward traversal and reverse traversal
1190// from a key that is not in the data. See method declartion for
1191// details.
1192template <typename Key, typename Value>
1193typename db<Key, Value>::iterator& db<Key, Value>::iterator::seek(
1194 art_key_type search_key, bool& match, bool fwd) {
1195 invalidate(); // invalidate the iterator (clear the stack).
1196 match = false; // unless we wind up with an exact match.
1197 if (UNODB_DETAIL_UNLIKELY(db_.root == nullptr)) return *this; // aka end()
1198
1199 auto node{db_.root};
1200 const auto k = search_key;
1201 auto remaining_key{k};
1202
1203 while (true) {
1204 const auto node_type = node.type();
1205 if (node_type == node_type::LEAF) {
1206 const auto* const leaf{node.template ptr<leaf_type*>()};
1207 push_leaf(node);
1208 const auto cmp_ = leaf->cmp(k);
1209 if (cmp_ == 0) {
1210 match = true;
1211 return *this;
1212 }
1213 if (fwd) { // GTE semantics
1214 // if search_key < leaf, use the leaf, else next().
1215 return (cmp_ < 0) ? *this : next();
1216 }
1217 // LTE semantics: if search_key > leaf, use the leaf, else prior().
1218 return (cmp_ > 0) ? *this : prior();
1219 }
1220 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
1221 auto* const inode{node.template ptr<inode_type*>()}; // some internal node.
1222 const auto key_prefix{inode->get_key_prefix().get_snapshot()}; // prefix
1223 const auto key_prefix_length{key_prefix.length()}; // length of that prefix
1224 const auto shared_length = key_prefix.get_shared_length(
1225 remaining_key.get_u64()); // #of prefix bytes matched.
1226 if (shared_length < key_prefix_length) {
1227 // We have visited an internal node whose prefix is longer than
1228 // the bytes in the key that we need to match. To figure out
1229 // whether the search key would be located before or after the
1230 // current internal node, we need to compare the respective key
1231 // spans lexicographically. Since we have [shared_length] bytes
1232 // in common, we know that the next byte will tell us the
1233 // relative ordering of the key vs the prefix. So now we compare
1234 // prefix and key and the first byte where they differ.
1235 const auto cmp_ = static_cast<int>(remaining_key[shared_length]) -
1236 static_cast<int>(key_prefix[shared_length]);
1237 UNODB_DETAIL_ASSERT(cmp_ != 0);
1238 if (fwd) {
1239 if (cmp_ < 0) {
1240 // FWD and the search key is ordered before this node. We
1241 // want the left-most leaf under the node.
1242 return left_most_traversal(node);
1243 }
1244 // FWD and the search key is ordered after this node. Right
1245 // most descent and then next().
1246 return right_most_traversal(node).next();
1247 }
1248 // reverse traversal
1249 if (cmp_ < 0) {
1250 // REV and the search key is ordered before this node. We
1251 // want the preceeding key.
1252 return left_most_traversal(node).prior();
1253 }
1254 // REV and the search key is ordered after this node.
1255 return right_most_traversal(node);
1256 }
1257 remaining_key.shift_right(key_prefix_length);
1258 const auto res = inode->find_child(node_type, remaining_key[0]);
1259 if (res.second == nullptr) {
1260 // We are on a key byte during the descent that is not mapped by
1261 // the current node. Where we go next depends on whether we are
1262 // doing forward or reverse traversal.
1263 if (fwd) {
1264 // FWD: Take the next child_index that is mapped in the data
1265 // and then do a left-most descent to land on the key that is
1266 // the immediate successor of the desired key in the data.
1267 //
1268 // Note: We are probing with a key byte which does not appear
1269 // in our list of keys (this was verified above) so this will
1270 // always be the index the first entry whose key byte is
1271 // greater-than the probe value and [false] if there is no
1272 // such entry.
1273 //
1274 // Note: [node] has not been pushed onto the stack yet!
1275 auto nxt = inode->gte_key_byte(node_type, remaining_key[0]);
1276 if (!nxt) {
1277 // Pop entries off the stack until we find one with a
1278 // right-sibling of the path we took to this node and then
1279 // do a left-most descent under that right-sibling. If there
1280 // is no such parent, we will wind up with an empty stack
1281 // (aka the end() iterator) and return that state.
1282 if (!empty()) pop();
1283 while (!empty()) {
1284 const auto& centry = top();
1285 const auto cnode{centry.node}; // possible parent from the stack
1286 auto* const icnode{cnode.template ptr<inode_type*>()};
1287 const auto cnxt = icnode->next(
1288 cnode.type(), centry.child_index); // right-sibling.
1289 if (cnxt) {
1290 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1291 return left_most_traversal(nchild);
1292 }
1293 pop();
1294 }
1295 return *this; // stack is empty (aka end()).
1296 }
1297 const auto& tmp = nxt.value(); // unwrap.
1298 const auto child_index = tmp.child_index;
1299 const auto child = inode->get_child(node_type, child_index);
1300 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1301 return left_most_traversal(child); // left most traversal
1302 }
1303 // REV: Take the prior child_index that is mapped and then do
1304 // a right-most descent to land on the key that is the
1305 // immediate precessor of the desired key in the data.
1306 auto nxt = inode->lte_key_byte(node_type, remaining_key[0]);
1307 if (!nxt) {
1308 // Pop off the current entry until we find one with a
1309 // left-sibling and then do a right-most descent under that
1310 // left-sibling. In the extreme case there is no such
1311 // previous entry and we will wind up with an empty stack.
1312 if (!empty()) pop();
1313 while (!empty()) {
1314 const auto& centry = top();
1315 const auto cnode{centry.node}; // possible parent from stack
1316 auto* const icnode{cnode.template ptr<inode_type*>()};
1317 const auto cnxt =
1318 icnode->prior(cnode.type(), centry.child_index); // left-sibling.
1319 if (cnxt) {
1320 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1321 return right_most_traversal(nchild);
1322 }
1323 pop();
1324 }
1325 return *this; // stack is empty (aka end()).
1326 }
1327 const auto& tmp = nxt.value(); // unwrap.
1328 const auto child_index{tmp.child_index};
1329 const auto child = inode->get_child(node_type, child_index);
1330 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1331 return right_most_traversal(child); // right most traversal
1332 }
1333 // Simple case. There is a child for the current key byte.
1334 const auto child_index{res.first};
1335 const auto* const child{res.second};
1336 push(node, remaining_key[0], child_index, key_prefix);
1337 node = *child;
1338 remaining_key.shift_right(1);
1339 } // while ( true )
1340 UNODB_DETAIL_CANNOT_HAPPEN();
1341}

1421template <typename Key, typename Value>
1422template <typename FN>
1423void db<Key, Value>::scan_range(Key from_key, Key to_key, FN fn) {
1424 // TODO(thompsonbry) : variable length keys. Explore a cheaper way
1425 // to handle the exclusive bound case when developing variable
1426 // length key support based on the maintained key buffer.
1427 constexpr bool debug = false; // set true to debug scan.

1429 const art_key_type to_key_{to_key}; // convert to internal key

1432 if (ret == 0) return; // NOP

1435 iterator it(*this);
