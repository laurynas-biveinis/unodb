370 void push(detail::node_ptr node, std::byte key_byte,
371 std::uint8_t child_index) {
372 // For variable length keys we need to know the number of bytes
373 // associated with the node's key_prefix. In addition there is
374 // one byte for the descent to the child node along the
375 // child_index. That information needs to be stored on the
376 // stack so we can pop off the right number of bytes even for
377 // OLC where the node might be concurrently modified.
378 UNODB_DETAIL_ASSERT(node.type() != node_type::LEAF);
379 auto* inode{node.ptr<detail::inode<Key>*>()};
380 auto prefix{inode->get_key_prefix().get_snapshot()};
381 stack_.push({{node, key_byte, child_index}, prefix.size()});
382 keybuf_.push(prefix.get_key_view());
383 keybuf_.push(key_byte);
384 }
385
387 void push_leaf(detail::node_ptr aleaf) {
388 // Mock up a stack entry for the leaf.
389 stack_.push({
390 {
391 aleaf,
392 static_cast<std::byte>(0xFFU), // ignored for leaf
393 static_cast<std::uint8_t>(0xFFU) // ignored for leaf
394 },
395 0 // ignored for leaf
396 });
397 }

842template <typename Key, class INode>
843detail::node_ptr* impl_helpers::add_or_choose_subtree(
844 INode& inode, std::byte key_byte, basic_art_key<Key> k, value_view v,
845 db<Key>& db_instance, tree_depth<basic_art_key<Key>> depth,
846 detail::node_ptr* node_in_parent) {
847 auto* const child =
848 unwrap_fake_critical_section(inode.find_child(key_byte).second);
849
850 if (child != nullptr) return child;
851
852 auto leaf = art_policy<Key>::make_db_leaf_ptr(k, v, db_instance);
853 const auto children_count = inode.get_children_count();
854
855 if constexpr (!std::is_same_v<INode, inode_256<Key>>) {
856 if (UNODB_DETAIL_UNLIKELY(children_count == INode::capacity)) {
857 auto larger_node{INode::larger_derived_type::create(
858 db_instance, inode, std::move(leaf), depth)};
859 *node_in_parent =
860 node_ptr{larger_node.release(), INode::larger_derived_type::type};
861#ifdef UNODB_DETAIL_WITH_STATS
862 db_instance
863 .template account_growing_inode<INode::larger_derived_type::type>();
864#endif // UNODB_DETAIL_WITH_STATS
865 return child;
866 }
867 }
868 inode.add_to_nonfull(std::move(leaf), depth, children_count);
869 return child;
870}

872template <typename Key, class INode>
873std::optional<detail::node_ptr*> impl_helpers::remove_or_choose_subtree(
874 INode& inode, std::byte key_byte, basic_art_key<Key> k,
875 db<Key>& db_instance, detail::node_ptr* node_in_parent) {
876 const auto [child_i, child_ptr]{inode.find_child(key_byte)};
877
878 if (child_ptr == nullptr) return {};
879
880 const auto child_ptr_val{child_ptr->load()};
881 if (child_ptr_val.type() != node_type::LEAF)
882 return unwrap_fake_critical_section(child_ptr);
883
884 const auto* const leaf{
885 child_ptr_val.template ptr<typename db<Key>::leaf_type*>()};
886 if (!leaf->matches(k)) return {};
887
888 if (UNODB_DETAIL_UNLIKELY(inode.is_min_size())) {
889 if constexpr (std::is_same_v<INode, inode_4<Key>>) {
890 auto current_node{
891 art_policy<Key>::make_db_inode_unique_ptr(&inode, db_instance)};
892 *node_in_parent = current_node->leave_last_child(child_i, db_instance);
893 } else {
894 auto new_node{
895 INode::smaller_derived_type::create(db_instance, inode, child_i)};
896 *node_in_parent =
897 node_ptr{new_node.release(), INode::smaller_derived_type::type};
898 }
899#ifdef UNODB_DETAIL_WITH_STATS
900 db_instance.template account_shrinking_inode<INode::type>();
901#endif // UNODB_DETAIL_WITH_STATS
902 return nullptr;
903 }
904
905 inode.remove(child_i, db_instance);
906 return nullptr;
907}

950template <class Key>
951bool db<Key>::insert_internal(art_key_type k, value_view v) {
952 if (UNODB_DETAIL_UNLIKELY(root == nullptr)) {
953 auto leaf = art_policy::make_db_leaf_ptr(k, v, *this);
954 root = detail::node_ptr{leaf.release(), node_type::LEAF};
955 return true;
956 }
957
958 auto* node = &root;
959 tree_depth_type depth{};
960 auto remaining_key{k};
961
962 while (true) {
963 const auto node_type = node->type();
964 if (node_type == node_type::LEAF) {
965 auto* const leaf{node->template ptr<leaf_type*>()};
966 const auto existing_key{leaf->get_key_view()};
967 if (UNODB_DETAIL_UNLIKELY(k.cmp(existing_key) == 0)) {
968 return false; // exists
969 }
970 // Replace the existing leaf with a new N4 and put the existing
971 // leaf and the a leaf for the caller's key and value under the
972 // new inode as its direct children.
973 auto new_leaf = art_policy::make_db_leaf_ptr(k, v, *this);
974 auto new_node{inode_4::create(*this, existing_key, remaining_key, depth,
975 leaf, std::move(new_leaf))};
976 *node = detail::node_ptr{new_node.release(), node_type::I4};
977#ifdef UNODB_DETAIL_WITH_STATS
978 account_growing_inode<node_type::I4>();
979#endif // UNODB_DETAIL_WITH_STATS
980 return true;
981 }
982
983 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
984
985 auto* const inode{node->template ptr<inode_type*>()};
986 const auto& key_prefix{inode->get_key_prefix()};
987 const auto key_prefix_length{key_prefix.length()};
988 const auto shared_prefix_len{key_prefix.get_shared_length(remaining_key)};
989 if (shared_prefix_len < key_prefix_length) {
990 // We have reached an existing inode whose key_prefix is greater
991 // than the desired match. We need to split this inode into a
992 // new N4 whose children are the existing inode and a new child
993 // leaf.
994 auto leaf = art_policy::make_db_leaf_ptr(k, v, *this);
995 auto new_node = inode_4::create(*this, *node, shared_prefix_len, depth,
996 std::move(leaf));
997 *node = detail::node_ptr{new_node.release(), node_type::I4};
998#ifdef UNODB_DETAIL_WITH_STATS
999 account_growing_inode<node_type::I4>();
1000 ++key_prefix_splits;
1001 UNODB_DETAIL_ASSERT(growing_inode_counts[internal_as_i<node_type::I4>] >
1002 key_prefix_splits);
1003#endif // UNODB_DETAIL_WITH_STATS
1004 return true;
1005 }
1006 // key_prefix bytes were absorbed during the descent. Now we need
1007 // to either descend along an existing child or create a new child.
1008 UNODB_DETAIL_ASSERT(shared_prefix_len == key_prefix_length);
1009 depth += key_prefix_length;
1010 remaining_key.shift_right(key_prefix_length);
1011
1012 node = inode->template add_or_choose_subtree<detail::node_ptr*>(
1013 node_type, remaining_key[0], k, v, *this, depth, node);
1014
1015 if (node == nullptr) return true;
1016
1017 ++depth;
1018 remaining_key.shift_right(1);
1019 }

1098// Position the iterator on the next leaf in the index.
1099template <class Key>
1100typename db<Key>::iterator& db<Key>::iterator::next() {
1101 while (!empty()) {
1102 auto e = top();
1103 auto node{e.node};
1104 UNODB_DETAIL_ASSERT(node != nullptr);
1105 auto node_type = node.type();
1106 if (node_type == node_type::LEAF) {
1107 pop(); // pop off the leaf
1108 continue; // falls through loop if just a root leaf since stack now
1109 // empty.
1110 }
1111 auto* inode{node.template ptr<inode_type*>()};
1112 auto nxt = inode->next(node_type,
1113 e.child_index); // next child of that parent.
1114 if (!nxt.has_value()) {
1115 pop(); // Nothing more for that inode.
1116 continue; // We will look for the right sibling of the parent inode.
1117 }
1118 // Fix up stack for new parent node state and left-most descent.
1119 auto e2 = nxt.value();
1120 pop();
1121 push(e2);
1122 auto child = inode->get_child(node_type, e2.child_index); // descend
1123 return left_most_traversal(child);
1124 }
1125 return *this; // stack is empty, so iterator == end().
1126}

1133 auto node{e.node};
1134 UNODB_DETAIL_ASSERT(node != nullptr);

1150 pop();
1151 push(e2);
1152 auto child = inode->get_child(node_type, e2.child_index); // descend
1153 return right_most_traversal(child);
1154 }
1155 return *this; // stack is empty, so iterator == end().

1159// node to the left-most leaf under that node, pushing nodes onto the
1160// stack as they are visited.
1161template <class Key>

1165 UNODB_DETAIL_ASSERT(node != nullptr);

1168 push_leaf(node);
1169 return *this; // done

1172 auto* const inode{node.ptr<inode_type*>()};

1174 push(e); // push the entry on the stack.
1175 node = inode->get_child(node_type, e.child_index); // get the child
1176 }
1177 UNODB_DETAIL_CANNOT_HAPPEN();

1202// Note: The basic seek() logic is similar to ::get() as long as the
1203// search_key exists in the data. However, the iterator is positioned
1204// instead of returning the value for the key. Life gets a lot more
1205// complicated when the search_key is not in the data and we have to
1206// consider the cases for both forward traversal and reverse traversal
1207// from a key that is not in the data. See method declartion for
1208// details.
1209template <class Key>
1210typename db<Key>::iterator& db<Key>::iterator::seek(art_key_type search_key,
1211 bool& match, bool fwd) {
1212 invalidate(); // invalidate the iterator (clear the stack).
1213 match = false; // unless we wind up with an exact match.
1214 if (UNODB_DETAIL_UNLIKELY(db_.root == nullptr)) return *this; // aka end()
1215
1216 auto node{db_.root};
1217 const auto k = search_key;
1218 auto remaining_key{k};
1219
1220 while (true) {
1221 const auto node_type = node.type();
1222 if (node_type == node_type::LEAF) {
1223 const auto* const leaf{node.template ptr<leaf_type*>()};
1224 push_leaf(node);
1225 const auto cmp_ = leaf->cmp(k);
1226 if (cmp_ == 0) {
1227 match = true;
1228 return *this;
1229 }
1230 if (fwd) { // GTE semantics
1231 // if search_key < leaf, use the leaf, else next().
1232 return (cmp_ < 0) ? *this : next();
1233 }
1234 // LTE semantics: if search_key > leaf, use the leaf, else prior().
1235 return (cmp_ > 0) ? *this : prior();
1236 }
1237 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
1238 auto* const inode{node.template ptr<inode_type*>()}; // some internal node.
1239 const auto& key_prefix{inode->get_key_prefix()}; // prefix for that node.
1240 const auto key_prefix_length{
1241 key_prefix.length()}; // length of that prefix.
1242 const auto shared_length = key_prefix.get_shared_length(
1243 remaining_key); // #of prefix bytes matched.
1244 if (shared_length < key_prefix_length) {
1245 // We have visited an internal node whose prefix is longer than
1246 // the bytes in the key that we need to match. To figure out
1247 // whether the search key would be located before or after the
1248 // current internal node, we need to compare the respective key
1249 // spans lexicographically. Since we have [shared_length] bytes
1250 // in common, we know that the next byte will tell us the
1251 // relative ordering of the key vs the prefix. So now we compare
1252 // prefix and key and the first byte where they differ.
1253 const auto cmp_ = static_cast<int>(remaining_key[shared_length]) -
1254 static_cast<int>(key_prefix[shared_length]);
1255 UNODB_DETAIL_ASSERT(cmp_ != 0);
1256 if (fwd) {
1257 if (cmp_ < 0) {
1258 // FWD and the search key is ordered before this node. We
1259 // want the left-most leaf under the node.
1260 return left_most_traversal(node);
1261 }
1262 // FWD and the search key is ordered after this node. Right
1263 // most descent and then next().
1264 return right_most_traversal(node).next();
1265 }
1266 // reverse traversal
1267 if (cmp_ < 0) {
1268 // REV and the search key is ordered before this node. We
1269 // want the preceeding key.
1270 return left_most_traversal(node).prior();
1271 }
1272 // REV and the search key is ordered after this node.
1273 return right_most_traversal(node);
1274 }
1275 remaining_key.shift_right(key_prefix_length);
1276 auto res = inode->find_child(node_type, remaining_key[0]);
1277 if (res.second == nullptr) {
1278 // We are on a key byte during the descent that is not mapped by
1279 // the current node. Where we go next depends on whether we are
1280 // doing forward or reverse traversal.
1281 if (fwd) {
1282 // FWD: Take the next child_index that is mapped in the data
1283 // and then do a left-most descent to land on the key that is
1284 // the immediate successor of the desired key in the data.
1285 //
1286 // Note: We are probing with a key byte which does not appear
1287 // in our list of keys (this was verified above) so this will
1288 // always be the index the first entry whose key byte is
1289 // greater-than the probe value and [false] if there is no
1290 // such entry.
1291 //
1292 // Note: [node] has not been pushed onto the stack yet!
1293 auto nxt = inode->gte_key_byte(node_type, remaining_key[0]);
1294 if (!nxt) {
1295 // Pop entries off the stack until we find one with a
1296 // right-sibling of the path we took to this node and then
1297 // do a left-most descent under that right-sibling. If there
1298 // is no such parent, we will wind up with an empty stack
1299 // (aka the end() iterator) and return that state.
1300 if (!empty()) pop();
1301 while (!empty()) {
1302 const auto centry = top();
1303 const auto cnode{centry.node}; // possible parent from the stack
1304 auto* const icnode{cnode.template ptr<inode_type*>()};
1305 const auto cnxt = icnode->next(
1306 cnode.type(), centry.child_index); // right-sibling.
1307 if (cnxt) {
1308 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1309 return left_most_traversal(nchild);
1310 }
1311 pop();
1312 }
1313 return *this; // stack is empty (aka end()).
1314 }
1315 auto tmp = nxt.value(); // unwrap.
1316 const auto child_index = tmp.child_index;
1317 const auto child = inode->get_child(node_type, child_index);
1318 push(node, tmp.key_byte, child_index); // the path we took
1319 return left_most_traversal(child); // left most traversal
1320 }
1321 // REV: Take the prior child_index that is mapped and then do
1322 // a right-most descent to land on the key that is the
1323 // immediate precessor of the desired key in the data.
1324 auto nxt = inode->lte_key_byte(node_type, remaining_key[0]);
1325 if (!nxt) {
1326 // Pop off the current entry until we find one with a
1327 // left-sibling and then do a right-most descent under that
1328 // left-sibling. In the extreme case there is no such
1329 // previous entry and we will wind up with an empty stack.
1330 if (!empty()) pop();
1331 while (!empty()) {
1332 const auto centry = top();
1333 const auto cnode{centry.node}; // possible parent from stack
1334 auto* const icnode{cnode.template ptr<inode_type*>()};
1335 const auto cnxt =
1336 icnode->prior(cnode.type(), centry.child_index); // left-sibling.
1337 if (cnxt) {
1338 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1339 return right_most_traversal(nchild);
1340 }
1341 pop();
1342 }
1343 return *this; // stack is empty (aka end()).
1344 }
1345 auto tmp = nxt.value(); // unwrap.
1346 const auto child_index{tmp.child_index};
1347 const auto child = inode->get_child(node_type, child_index);
1348 push(node, tmp.key_byte, child_index); // the path we took
1349 return right_most_traversal(child); // right most traversal
1350 }
1351 // Simple case. There is a child for the current key byte.
1352 const auto child_index{res.first};
1353 const auto* const child{res.second};
1354 push(node, remaining_key[0], child_index);
1355 node = *child;
1356 remaining_key.shift_right(1);
1357 } // while ( true )
1358 UNODB_DETAIL_CANNOT_HAPPEN();
1359}
