335 void push(detail::node_ptr node, std::byte key_byte,
336 std::uint8_t child_index, detail::key_prefix_snapshot prefix) {
337 // For variable length keys we need to know the number of bytes
338 // associated with the node's key_prefix. In addition there is
339 // one byte for the descent to the child node along the
340 // child_index. That information needs to be stored on the
341 // stack so we can pop off the right number of bytes even for
342 // OLC where the node might be concurrently modified.
343 UNODB_DETAIL_ASSERT(node.type() != node_type::LEAF);
344 stack_.push({node, key_byte, child_index, prefix});
345 keybuf_.push(prefix.get_key_view());
346 keybuf_.push(key_byte);
347 }
348
350 void push_leaf(detail::node_ptr aleaf) {
351 // Mock up a stack entry for the leaf.
352 stack_.push({
353 aleaf,
354 static_cast<std::byte>(0xFFU), // ignored for leaf
355 static_cast<std::uint8_t>(0xFFU), // ignored for leaf
356 detail::key_prefix_snapshot(0) // ignored for leaf
357 });
358 // No change in the key_buffer.
359 }
360
362 void push(const typename inode_base::iter_result& e) {
363 const auto node_type = e.node.type();
364 if (UNODB_DETAIL_UNLIKELY(node_type == node_type::LEAF)) {
365 push_leaf(e.node);
366 }
367 push(e.node, e.key_byte, e.child_index, e.prefix);
368 }

371 void pop() {
372 const auto prefix_len = top().prefix.length();
373 keybuf_.pop(prefix_len);
374 stack_.pop();
375 }

799template <typename Key, class INode>
800detail::node_ptr* impl_helpers::add_or_choose_subtree(
801 INode& inode, std::byte key_byte, basic_art_key<Key> k, value_view v,
802 db<Key>& db_instance, tree_depth<basic_art_key<Key>> depth,
803 detail::node_ptr* node_in_parent) {
804 auto* const child =
805 unwrap_fake_critical_section(inode.find_child(key_byte).second);
806
807 if (child != nullptr) return child;
808
809 auto leaf = art_policy<Key>::make_db_leaf_ptr(k, v, db_instance);
810 const auto children_count = inode.get_children_count();
811
812 if constexpr (!std::is_same_v<INode, inode_256<Key>>) {
813 if (UNODB_DETAIL_UNLIKELY(children_count == INode::capacity)) {
814 auto larger_node{INode::larger_derived_type::create(
815 db_instance, inode, std::move(leaf), depth)};
816 *node_in_parent =
817 node_ptr{larger_node.release(), INode::larger_derived_type::type};
818#ifdef UNODB_DETAIL_WITH_STATS
819 db_instance
820 .template account_growing_inode<INode::larger_derived_type::type>();
821#endif // UNODB_DETAIL_WITH_STATS
822 return child;
823 }
824 }
825 inode.add_to_nonfull(std::move(leaf), depth, children_count);
826 return child;
827}

829template <typename Key, class INode>
830std::optional<detail::node_ptr*> impl_helpers::remove_or_choose_subtree(
831 INode& inode, std::byte key_byte, basic_art_key<Key> k,
832 db<Key>& db_instance, detail::node_ptr* node_in_parent) {
833 const auto [child_i, child_ptr]{inode.find_child(key_byte)};
834
835 if (child_ptr == nullptr) return {};
836
837 const auto child_ptr_val{child_ptr->load()};
838 if (child_ptr_val.type() != node_type::LEAF)
839 return unwrap_fake_critical_section(child_ptr);
840
841 const auto* const leaf{
842 child_ptr_val.template ptr<typename db<Key>::leaf_type*>()};
843 if (!leaf->matches(k)) return {};
844
845 if (UNODB_DETAIL_UNLIKELY(inode.is_min_size())) {
846 if constexpr (std::is_same_v<INode, inode_4<Key>>) {
847 auto current_node{
848 art_policy<Key>::make_db_inode_unique_ptr(&inode, db_instance)};
849 *node_in_parent = current_node->leave_last_child(child_i, db_instance);
850 } else {
851 auto new_node{
852 INode::smaller_derived_type::create(db_instance, inode, child_i)};
853 *node_in_parent =
854 node_ptr{new_node.release(), INode::smaller_derived_type::type};
855 }
856#ifdef UNODB_DETAIL_WITH_STATS
857 db_instance.template account_shrinking_inode<INode::type>();
858#endif // UNODB_DETAIL_WITH_STATS
859 return nullptr;
860 }
861
862 inode.remove(child_i, db_instance);
863 return nullptr;
864}

881 while (true) {
882 const auto node_type = node.type();
883 if (node_type == node_type::LEAF) {
884 const auto* const leaf{node.template ptr<leaf_type*>()};
885 if (leaf->matches(k)) return leaf->get_value_view();
886 return {};
887 }
888
889 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
890
891 auto* const inode{node.template ptr<inode_type*>()};
892 const auto& key_prefix{inode->get_key_prefix()};
893 const auto key_prefix_length{key_prefix.length()};
894 if (key_prefix.get_shared_length(remaining_key) < key_prefix_length)
895 return {};
896 remaining_key.shift_right(key_prefix_length);
897 const auto* const child{
898 inode->find_child(node_type, remaining_key[0]).second};
899 if (child == nullptr) return {};
900
901 node = *child;
902 remaining_key.shift_right(1);
903 }

907template <typename Key>
908bool db<Key>::insert_internal(art_key_type k, value_view v) {
909 if (UNODB_DETAIL_UNLIKELY(root == nullptr)) {
910 auto leaf = art_policy::make_db_leaf_ptr(k, v, *this);
911 root = detail::node_ptr{leaf.release(), node_type::LEAF};
912 return true;
913 }
914
915 auto* node = &root;
916 tree_depth_type depth{};
917 auto remaining_key{k};
918
919 while (true) {
920 const auto node_type = node->type();
921 if (node_type == node_type::LEAF) {
922 auto* const leaf{node->template ptr<leaf_type*>()};
923 const auto existing_key{leaf->get_key_view()};
924 const auto cmp = k.cmp(existing_key);
925 if (UNODB_DETAIL_UNLIKELY(cmp == 0)) {
926 return false; // exists
927 }
928 // Replace the existing leaf with a new N4 and put the existing
929 // leaf and the leaf for the caller's key and value under the
930 // new inode as its direct children.
931 auto new_leaf = art_policy::make_db_leaf_ptr(k, v, *this);
932 auto new_node{inode_4::create(*this, existing_key, remaining_key, depth,
933 leaf, std::move(new_leaf))};
934 *node = detail::node_ptr{new_node.release(), node_type::I4};
935#ifdef UNODB_DETAIL_WITH_STATS
936 account_growing_inode<node_type::I4>();
937#endif // UNODB_DETAIL_WITH_STATS
938 return true;
939 }
940
941 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
942
943 auto* const inode{node->template ptr<inode_type*>()};
944 const auto& key_prefix{inode->get_key_prefix()};
945 const auto key_prefix_length{key_prefix.length()};
946 const auto shared_prefix_len{key_prefix.get_shared_length(remaining_key)};
947 if (shared_prefix_len < key_prefix_length) {
948 // We have reached an existing inode whose key_prefix is greater
949 // than the desired match. We need to split this inode into a
950 // new N4 whose children are the existing inode and a new child
951 // leaf.
952 auto leaf = art_policy::make_db_leaf_ptr(k, v, *this);
953 auto new_node = inode_4::create(*this, *node, shared_prefix_len, depth,
954 std::move(leaf));
955 *node = detail::node_ptr{new_node.release(), node_type::I4};
956#ifdef UNODB_DETAIL_WITH_STATS
957 account_growing_inode<node_type::I4>();
958 ++key_prefix_splits;
959 UNODB_DETAIL_ASSERT(growing_inode_counts[internal_as_i<node_type::I4>] >
960 key_prefix_splits);
961#endif // UNODB_DETAIL_WITH_STATS
962 return true;
963 }
964 // key_prefix bytes were absorbed during the descent. Now we need
965 // to either descend along an existing child or create a new child.
966 UNODB_DETAIL_ASSERT(shared_prefix_len == key_prefix_length);
967 depth += key_prefix_length;
968 remaining_key.shift_right(key_prefix_length);
969
970 node = inode->template add_or_choose_subtree<detail::node_ptr*>(
971 node_type, remaining_key[0], k, v, *this, depth, node);
972
973 if (node == nullptr) return true;
974
975 ++depth;
976 remaining_key.shift_right(1);
977 }

1031// Traverse to the left-most leaf. The stack is cleared first and then
1032// re-populated as we step down along the path to the left-most leaf.
1033// If the tree is empty, then the result is the same as end().
1034// TODO(laurynas): the method pairs first, last; next, prior;
1035// left_most_traversal, right_most_traversal are identical except for a couple
1036// lines. Extract helper methods templatized on the differences.

1038typename db<Key>::iterator& db<Key>::iterator::first() {
1039 invalidate(); // clear the stack
1040 if (UNODB_DETAIL_UNLIKELY(db_.root == nullptr)) return *this; // empty tree.
1041 auto node{db_.root};
1042 return left_most_traversal(node);
1043}

1046// re-populated as we step down along the path to the right-most leaf.
1047// If the tree is empty, then the result is the same as end().

1049typename db<Key>::iterator& db<Key>::iterator::last() {
1050 invalidate(); // clear the stack
1051 if (UNODB_DETAIL_UNLIKELY(db_.root == nullptr)) return *this; // empty tree.
1052 auto node{db_.root};

1056// Position the iterator on the next leaf in the index.
1057template <typename Key>
1058typename db<Key>::iterator& db<Key>::iterator::next() {
1059 while (!empty()) {
1060 auto e = top();
1061 auto node{e.node};
1062 UNODB_DETAIL_ASSERT(node != nullptr);
1063 auto node_type = node.type();
1064 if (node_type == node_type::LEAF) {
1065 pop(); // pop off the leaf
1066 continue; // falls through loop if just a root leaf since stack now
1067 // empty.
1068 }
1069 auto* inode{node.template ptr<inode_type*>()};
1070 auto nxt = inode->next(node_type,
1071 e.child_index); // next child of that parent.
1072 if (!nxt.has_value()) {
1073 pop(); // Nothing more for that inode.
1074 continue; // We will look for the right sibling of the parent inode.
1075 }
1076 // Fix up stack for new parent node state and left-most descent.
1077 auto e2 = nxt.value();
1078 pop();
1079 push(e2);
1080 auto child = inode->get_child(node_type, e2.child_index); // descend
1081 return left_most_traversal(child);
1082 }
1083 return *this; // stack is empty, so iterator == end().
1084}

1086// Position the iterator on the prior leaf in the index.
1087template <typename Key>
1088typename db<Key>::iterator& db<Key>::iterator::prior() {
1089 while (!empty()) {
1090 auto e = top();
1091 auto node{e.node};
1092 UNODB_DETAIL_ASSERT(node != nullptr);
1093 auto node_type = node.type();
1094 if (node_type == node_type::LEAF) {
1095 pop(); // pop off the leaf
1096 continue; // falls through loop if just a root leaf since stack now
1097 // empty.
1098 }
1099 auto* inode{node.template ptr<inode_type*>()};
1100 auto nxt = inode->prior(node_type, e.child_index); // parent's prev child
1101 if (!nxt) {
1102 pop(); // Nothing more for that inode.
1103 continue; // We will look for the left sibling of the parent inode.
1104 }
1105 // Fix up stack for new parent node state and right-most descent.
1106 UNODB_DETAIL_ASSERT(nxt.has_value()); // value exists for std::optional
1107 auto e2 = nxt.value();
1108 pop();
1109 push(e2);
1110 auto child = inode->get_child(node_type, e2.child_index); // descend
1111 return right_most_traversal(child);
1112 }
1113 return *this; // stack is empty, so iterator == end().
1114}

1117// node to the left-most leaf under that node, pushing nodes onto the
1118// stack as they are visited.
1119template <typename Key>

1121 detail::node_ptr node) {
1122 while (true) {
1123 UNODB_DETAIL_ASSERT(node != nullptr);
1124 const auto node_type = node.type();

1126 push_leaf(node);
1127 return *this; // done

1160// Note: The basic seek() logic is similar to ::get() as long as the
1161// search_key exists in the data. However, the iterator is positioned
1162// instead of returning the value for the key. Life gets a lot more
1163// complicated when the search_key is not in the data and we have to
1164// consider the cases for both forward traversal and reverse traversal
1165// from a key that is not in the data. See method declartion for
1166// details.
1167template <typename Key>
1168typename db<Key>::iterator& db<Key>::iterator::seek(art_key_type search_key,
1169 bool& match, bool fwd) {
1170 invalidate(); // invalidate the iterator (clear the stack).
1171 match = false; // unless we wind up with an exact match.
1172 if (UNODB_DETAIL_UNLIKELY(db_.root == nullptr)) return *this; // aka end()
1173
1174 auto node{db_.root};
1175 const auto k = search_key;
1176 auto remaining_key{k};
1177
1178 while (true) {
1179 const auto node_type = node.type();
1180 if (node_type == node_type::LEAF) {
1181 const auto* const leaf{node.template ptr<leaf_type*>()};
1182 push_leaf(node);
1183 const auto cmp_ = leaf->cmp(k);
1184 if (cmp_ == 0) {
1185 match = true;
1186 return *this;
1187 }
1188 if (fwd) { // GTE semantics
1189 // if search_key < leaf, use the leaf, else next().
1190 return (cmp_ < 0) ? *this : next();
1191 }
1192 // LTE semantics: if search_key > leaf, use the leaf, else prior().
1193 return (cmp_ > 0) ? *this : prior();
1194 }
1195 UNODB_DETAIL_ASSERT(node_type != node_type::LEAF);
1196 auto* const inode{node.template ptr<inode_type*>()}; // some internal node.
1197 const auto key_prefix{inode->get_key_prefix().get_snapshot()}; // prefix
1198 const auto key_prefix_length{key_prefix.length()}; // length of that prefix
1199 const auto shared_length = key_prefix.get_shared_length(
1200 remaining_key.get_u64()); // #of prefix bytes matched.
1201 if (shared_length < key_prefix_length) {
1202 // We have visited an internal node whose prefix is longer than
1203 // the bytes in the key that we need to match. To figure out
1204 // whether the search key would be located before or after the
1205 // current internal node, we need to compare the respective key
1206 // spans lexicographically. Since we have [shared_length] bytes
1207 // in common, we know that the next byte will tell us the
1208 // relative ordering of the key vs the prefix. So now we compare
1209 // prefix and key and the first byte where they differ.
1210 const auto cmp_ = static_cast<int>(remaining_key[shared_length]) -
1211 static_cast<int>(key_prefix[shared_length]);
1212 UNODB_DETAIL_ASSERT(cmp_ != 0);
1213 if (fwd) {
1214 if (cmp_ < 0) {
1215 // FWD and the search key is ordered before this node. We
1216 // want the left-most leaf under the node.
1217 return left_most_traversal(node);
1218 }
1219 // FWD and the search key is ordered after this node. Right
1220 // most descent and then next().
1221 return right_most_traversal(node).next();
1222 }
1223 // reverse traversal
1224 if (cmp_ < 0) {
1225 // REV and the search key is ordered before this node. We
1226 // want the preceeding key.
1227 return left_most_traversal(node).prior();
1228 }
1229 // REV and the search key is ordered after this node.
1230 return right_most_traversal(node);
1231 }
1232 remaining_key.shift_right(key_prefix_length);
1233 auto res = inode->find_child(node_type, remaining_key[0]);
1234 if (res.second == nullptr) {
1235 // We are on a key byte during the descent that is not mapped by
1236 // the current node. Where we go next depends on whether we are
1237 // doing forward or reverse traversal.
1238 if (fwd) {
1239 // FWD: Take the next child_index that is mapped in the data
1240 // and then do a left-most descent to land on the key that is
1241 // the immediate successor of the desired key in the data.
1242 //
1243 // Note: We are probing with a key byte which does not appear
1244 // in our list of keys (this was verified above) so this will
1245 // always be the index the first entry whose key byte is
1246 // greater-than the probe value and [false] if there is no
1247 // such entry.
1248 //
1249 // Note: [node] has not been pushed onto the stack yet!
1250 auto nxt = inode->gte_key_byte(node_type, remaining_key[0]);
1251 if (!nxt) {
1252 // Pop entries off the stack until we find one with a
1253 // right-sibling of the path we took to this node and then
1254 // do a left-most descent under that right-sibling. If there
1255 // is no such parent, we will wind up with an empty stack
1256 // (aka the end() iterator) and return that state.
1257 if (!empty()) pop();
1258 while (!empty()) {
1259 const auto centry = top();
1260 const auto cnode{centry.node}; // possible parent from the stack
1261 auto* const icnode{cnode.template ptr<inode_type*>()};
1262 const auto cnxt = icnode->next(
1263 cnode.type(), centry.child_index); // right-sibling.
1264 if (cnxt) {
1265 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1266 return left_most_traversal(nchild);
1267 }
1268 pop();
1269 }
1270 return *this; // stack is empty (aka end()).
1271 }
1272 auto tmp = nxt.value(); // unwrap.
1273 const auto child_index = tmp.child_index;
1274 const auto child = inode->get_child(node_type, child_index);
1275 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1276 return left_most_traversal(child); // left most traversal
1277 }
1278 // REV: Take the prior child_index that is mapped and then do
1279 // a right-most descent to land on the key that is the
1280 // immediate precessor of the desired key in the data.
1281 auto nxt = inode->lte_key_byte(node_type, remaining_key[0]);
1282 if (!nxt) {
1283 // Pop off the current entry until we find one with a
1284 // left-sibling and then do a right-most descent under that
1285 // left-sibling. In the extreme case there is no such
1286 // previous entry and we will wind up with an empty stack.
1287 if (!empty()) pop();
1288 while (!empty()) {
1289 const auto centry = top();
1290 const auto cnode{centry.node}; // possible parent from stack
1291 auto* const icnode{cnode.template ptr<inode_type*>()};
1292 const auto cnxt =
1293 icnode->prior(cnode.type(), centry.child_index); // left-sibling.
1294 if (cnxt) {
1295 auto nchild = icnode->get_child(cnode.type(), centry.child_index);
1296 return right_most_traversal(nchild);
1297 }
1298 pop();
1299 }
1300 return *this; // stack is empty (aka end()).
1301 }
1302 auto tmp = nxt.value(); // unwrap.
1303 const auto child_index{tmp.child_index};
1304 const auto child = inode->get_child(node_type, child_index);
1305 push(node, tmp.key_byte, child_index, tmp.prefix); // the path we took
1306 return right_most_traversal(child); // right most traversal
1307 }
1308 // Simple case. There is a child for the current key byte.
1309 const auto child_index{res.first};
1310 const auto* const child{res.second};
1311 push(node, remaining_key[0], child_index, key_prefix);
1312 node = *child;
1313 remaining_key.shift_right(1);
1314 } // while ( true )
1315 UNODB_DETAIL_CANNOT_HAPPEN();
1316}
