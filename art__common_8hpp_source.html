143template <typename T>
144constexpr typename std::enable_if<std::is_integral<T>::value, T>::type
145shift_or_32bit_int(T i) {
146 i |= (i >> 1);
147 i |= (i >> 2);
148 i |= (i >> 4);
149 i |= (i >> 8);
150 i |= (i >> 16);
151 return i;
152}
153
158template <typename T>
159[[nodiscard, gnu::pure]] constexpr
160 typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 4,
161 T>::type
162 next_power_of_two(T i) noexcept {
163 return shift_or_32bit_int(i) + static_cast<T>(1);
164}
165
166template <typename T>
167[[nodiscard, gnu::pure]] constexpr
168 typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 8,
169 T>::type
170 next_power_of_two(T i) noexcept {
171 i = shift_or_32bit_int(i);
172 i |= (i >> 32U);
173 return ++i;
174}

193template <typename T>
194[[nodiscard, gnu::pure]] constexpr T lexicographic_successor(T v) noexcept {
195 const T t = v | (v - 1u); // t gets v's least significant 0 bits set to 1
196 // Next set to 1 the most significant bit to change, set to 0 the
197 // least significant ones, and add the necessary 1 bits.
198 const T w = (t + 1) | (((~t & -~t) - 1) >> (ctz(v) + 1));
199 return w;
200}

212inline void ensure_capacity(std::byte *&buf, size_t &cap, size_t off,
213 size_t min_capacity) {
214 // Find the allocation size in bytes which satisfies that minimum capacity.
215 // We first look for the next power of two. Then we adjust for the case where
216 // the [min_capacity] is already a power of two (a common edge case).
217 auto nsize = detail::next_power_of_two(min_capacity);
218 auto asize = (min_capacity == (nsize >> 1U)) ? min_capacity : nsize;
219 auto *tmp = detail::allocate_aligned(asize); // new allocation.
220 std::memcpy(tmp, buf, off); // copy over the data.
221 if (cap > INITIAL_BUFFER_CAPACITY) { // free old buffer iff allocated
222 detail::free_aligned(buf);
223 }
224 buf = static_cast<std::byte *>(tmp);
225 cap = asize;
226}

456 key_encoder &encode_text(std::span<const std::byte> text) {
457 // truncate view to at most maxlen bytes.
458 text = (text.size_bytes() > maxlen) ? text.subspan(0, maxlen) : text;
459 // normalize padding by stripping off any trailing [pad] bytes.
460 auto sz = text.size_bytes();
461 for (; sz > 0; sz--) {
462 if (text[sz - 1] != pad) break;
463 }
464 text = text.subspan(0, sz); // adjust span in case truncated.
465 // Ensure enough room for the text, the pad byte, and the
466 // run-length encoding of the pad byte.
467 ensure_available(sz + 1 + sizeof(size_type));
468 const auto padlen{static_cast<size_type>(maxlen - sz)};
469 append_bytes(text); // append bytes to the buffer.
470 encode(static_cast<std::uint8_t>(pad)); // encode as unsigned byte
471 encode(padlen); // logical run-length of the pad byte.
472 return *this;
473 }
474
476 key_encoder &encode_text(std::string_view sv) {
477 return encode_text(std::span<const std::byte>(
478 reinterpret_cast<const std::byte *>(sv.data()), sv.size()));
479 }

536 key_decoder &decode(std::int8_t &v) noexcept {
537 constexpr auto one = static_cast<std::uint8_t>(1);
538 std::uint8_t u;
539 decode(u);
540 v = (u >= msb8) ? static_cast<int8_t>(u - msb8)
541 : -static_cast<int8_t>(msb8 - one - u) -
542 static_cast<std::int8_t>(1);
543 return *this;
544 }

552 : -static_cast<int16_t>(msb16 - one - u) -
553 static_cast<std::int16_t>(1);
554 return *this;
555 }
