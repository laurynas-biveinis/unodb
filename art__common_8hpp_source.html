138template <typename T>
139constexpr typename std::enable_if<std::is_integral<T>::value, T>::type
140shift_or_32bit_int(T i) {
141 i |= (i >> 1);
142 i |= (i >> 2);
143 i |= (i >> 4);
144 i |= (i >> 8);
145 i |= (i >> 16);
146 return i;
147}

153template <typename T>
154[[nodiscard, gnu::pure]] constexpr
155 typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 4,
156 T>::type
157 next_power_of_two(T i) noexcept {
158 return shift_or_32bit_int(i) + static_cast<T>(1);
159}
160
161template <typename T>
162[[nodiscard, gnu::pure]] constexpr
163 typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 8,
164 T>::type
165 next_power_of_two(T i) noexcept {
166 i = shift_or_32bit_int(i);
167 i |= (i >> 32U);
168 return ++i;
169}

188template <typename T>
189[[nodiscard, gnu::pure]] constexpr T lexicographic_successor(T v) noexcept {
190 const T t = v | (v - 1u); // t gets v's least significant 0 bits set to 1
191 // Next set to 1 the most significant bit to change, set to 0 the
192 // least significant ones, and add the necessary 1 bits.
193 const T w = (t + 1) | (((~t & -~t) - 1) >> (ctz(v) + 1));
194 return w;
195}

207inline void ensure_capacity(std::byte *&buf, size_t &cap, size_t off,
208 size_t min_capacity) {
209 // Find the allocation size in bytes which satisfies that minimum capacity.
210 // We first look for the next power of two. Then we adjust for the case where
211 // the [min_capacity] is already a power of two (a common edge case).
212 auto nsize = detail::next_power_of_two(min_capacity);
213 auto asize = (min_capacity == (nsize >> 1U)) ? min_capacity : nsize;
214 auto *tmp = detail::allocate_aligned(asize); // new allocation.
215 std::memcpy(tmp, buf, off); // copy over the data.
216 if (cap > INITIAL_BUFFER_CAPACITY) { // free old buffer iff allocated
217 detail::free_aligned(buf);
218 }
219 buf = static_cast<std::byte *>(tmp);
220 cap = asize;
221}

467 key_encoder &encode_text(std::span<const std::byte> text) {
468 // truncate view to at most maxlen bytes.
469 text = (text.size_bytes() > maxlen) ? text.subspan(0, maxlen) : text;
470 // normalize padding by stripping off any trailing [pad] bytes.
471 auto sz = text.size_bytes();
472 for (; sz > 0; sz--) {
473 if (text[sz - 1] != pad) break;
474 }
475 text = text.subspan(0, sz); // adjust span in case truncated.
476 // Ensure enough room for the text, the pad byte, and the
477 // run-length encoding of the pad byte.
478 ensure_available(sz + 1 + sizeof(size_type));
479 const auto padlen{static_cast<size_type>(maxlen - sz)};
480 append_bytes(text); // append bytes to the buffer.
481 encode(static_cast<std::uint8_t>(pad)); // encode as unsigned byte
482 encode(padlen); // logical run-length of the pad byte.
483 return *this;
484 }

574 : -static_cast<int32_t>(msb32 - one - u) - 1;

584 : -static_cast<int64_t>(msb64 - one - u) - 1LL;
