151template <typename T>
152[[nodiscard, gnu::pure]] constexpr T lexicographic_successor(T v) noexcept {
153 const T t = v | (v - 1u); // t gets v's least significant 0 bits set to 1
154 // Next set to 1 the most significant bit to change, set to 0 the
155 // least significant ones, and add the necessary 1 bits.
156 const T w = (t + 1) | (((~t & -~t) - 1) >> (std::countr_zero(v) + 1));
157 return w;
158}

170inline void ensure_capacity(std::byte *&buf, size_t &cap, size_t off,
171 size_t min_capacity) {
172 // Find the smallest power of two >= min_capacity.
173 const auto asize = std::bit_ceil(min_capacity);
174 auto *tmp = detail::allocate_aligned(asize); // new allocation.
175 std::memcpy(tmp, buf, off); // copy over the data.
176 if (cap > INITIAL_BUFFER_CAPACITY) { // free old buffer iff allocated
177 detail::free_aligned(buf);
178 }
179 buf = static_cast<std::byte *>(tmp);
180 cap = asize;
181}

420 key_encoder &encode_text(std::span<const std::byte> text) {
421 // truncate view to at most maxlen bytes.
422 text = (text.size_bytes() > maxlen) ? text.subspan(0, maxlen) : text;
423 // normalize padding by stripping off any trailing [pad] bytes.
424 auto sz = text.size_bytes();
425 for (; sz > 0; sz--) {
426 if (text[sz - 1] != pad) break;
427 }
428 text = text.subspan(0, sz); // adjust span in case truncated.
429 // Ensure enough room for the text, the pad byte, and the
430 // run-length encoding of the pad byte.
431 ensure_available(sz + 1 + sizeof(size_type));
432 const auto padlen{static_cast<size_type>(maxlen - sz)};
433 append_bytes(text); // append bytes to the buffer.
434 encode(static_cast<std::uint8_t>(pad)); // encode as unsigned byte
435 encode(padlen); // logical run-length of the pad byte.
436 return *this;
437 }

512 key_decoder &decode(std::int8_t &v) noexcept {
513 std::uint8_t u;
514 decode(u);
515 v = static_cast<std::int8_t>(
516 (u >= msb8) ? u - msb8 : ~static_cast<std::uint8_t>(msb8 - 1U - u));
517 return *this;
518 }

521 key_decoder &decode(std::int16_t &v) noexcept {
522 std::uint16_t u;
523 decode(u);
524 v = static_cast<std::int16_t>(
525 (u >= msb16) ? u - msb16 : ~static_cast<std::uint16_t>(msb16 - 1U - u));
526 return *this;
527 }

530 key_decoder &decode(std::int32_t &v) noexcept {
531 constexpr auto one = static_cast<std::uint32_t>(1);
532 std::uint32_t u;
533 decode(u);
534 v = (u >= msb32) ? static_cast<int32_t>(u - msb32)
535 : -static_cast<int32_t>(msb32 - one - u) - 1;
536 return *this;
537 }

543 decode(u);
544 v = (u >= msb64) ? static_cast<int64_t>(u - msb64)
545 : -static_cast<int64_t>(msb64 - one - u) - 1LL;
546 return *this;
