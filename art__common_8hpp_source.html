142constexpr typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 4,
143 T>::type
144next_power_of_two(T i) {
145 return shift_or_32bit_int(i) + static_cast<T>(1);

149constexpr typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 8,
150 T>::type
151next_power_of_two(T i) {
152 i = shift_or_32bit_int(i);
153 i |= (i >> 32U);
154 return ++i;

159inline void ensure_capacity(std::byte *&buf, // buffer to resize
160 size_t &cap, // current buffer capacity
161 size_t off, // current #of used bytes
162 size_t min_capacity // desired new minimum capacity
163) {
164 // Find the allocation size in bytes which satisfies that minimum
165 // capacity. We first look for the next power of two. Then we
166 // adjust for the case where the [min_capacity] is already a power
167 // of two (a common edge case).
168 auto nsize = detail::next_power_of_two(min_capacity);
169 auto asize = (min_capacity == (nsize >> 1U)) ? min_capacity : nsize;
170 auto *tmp = detail::allocate_aligned(asize); // new allocation.
171 std::memcpy(tmp, buf, off); // copy over the data.
172 if (cap > INITIAL_BUFFER_CAPACITY) { // free old buffer iff allocated
173 detail::free_aligned(buf);
174 }
175 buf = reinterpret_cast<std::byte *>(tmp);
176 cap = asize;
177}

379 key_decoder &decode(std::int8_t &v) {
380 const auto one = static_cast<std::uint8_t>(1);
381 std::uint8_t u;
382 decode(u);
383 v = (u >= msb8) ? static_cast<int8_t>(u - msb8)
384 : -static_cast<int8_t>(msb8 - one - u) -
385 static_cast<std::int8_t>(1);
386 return *this;
