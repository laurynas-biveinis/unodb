133template <typename T>
134constexpr typename std::enable_if<std::is_integral<T>::value, T>::type
135shift_or_32bit_int(T i) {
136 i |= (i >> 1);
137 i |= (i >> 2);
138 i |= (i >> 4);
139 i |= (i >> 8);
140 i |= (i >> 16);
141 return i;
142}

148template <typename T>
149constexpr typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 4,
150 T>::type
151next_power_of_two(T i) {
152 return shift_or_32bit_int(i) + static_cast<T>(1);
153}

155template <typename T>
156constexpr typename std::enable_if<std::is_integral<T>::value && sizeof(T) == 8,
157 T>::type
158next_power_of_two(T i) {
159 i = shift_or_32bit_int(i);
160 i |= (i >> 32U);
161 return ++i;
162}

166inline void ensure_capacity(std::byte *&buf, // buffer to resize
167 size_t &cap, // current buffer capacity
168 size_t off, // current #of used bytes
169 size_t min_capacity // desired new minimum capacity
170) {
171 // Find the allocation size in bytes which satisfies that minimum
172 // capacity. We first look for the next power of two. Then we
173 // adjust for the case where the [min_capacity] is already a power
174 // of two (a common edge case).
175 auto nsize = detail::next_power_of_two(min_capacity);
176 auto asize = (min_capacity == (nsize >> 1U)) ? min_capacity : nsize;
177 auto *tmp = detail::allocate_aligned(asize); // new allocation.
178 std::memcpy(tmp, buf, off); // copy over the data.
179 if (cap > INITIAL_BUFFER_CAPACITY) { // free old buffer iff allocated
180 detail::free_aligned(buf);
181 }
182 buf = reinterpret_cast<std::byte *>(tmp);
183 cap = asize;
184}

386 key_decoder &decode(std::int8_t &v) {
387 const auto one = static_cast<std::uint8_t>(1);
388 std::uint8_t u;
389 decode(u);
390 v = (u >= msb8) ? static_cast<int8_t>(u - msb8)
391 : -static_cast<int8_t>(msb8 - one - u) -
392 static_cast<std::int8_t>(1);
393 return *this;
394 }

404 return *this;
