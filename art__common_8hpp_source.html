165inline void ensure_capacity(std::byte*& buf, size_t& cap, size_t off,
166 size_t min_capacity) {
167 // Find the smallest power of two >= min_capacity.
168 const auto asize = std::bit_ceil(min_capacity);
169 auto* tmp = detail::allocate_aligned(asize); // new allocation.
170 std::memcpy(tmp, buf, off); // copy over the data.
171 if (cap > INITIAL_BUFFER_CAPACITY) { // free old buffer iff allocated
172 detail::free_aligned(buf);
173 }
174 buf = static_cast<std::byte*>(tmp);
175 cap = asize;
176}

471 key_encoder& encode_text(std::span<const std::byte> text) {
472 // truncate view to at most maxlen bytes.
473 text = (text.size_bytes() > maxlen) ? text.subspan(0, maxlen) : text;
474 // normalize padding by stripping off any trailing [pad] bytes.
475 auto sz = text.size_bytes();
476 for (; sz > 0; sz--) {
477 if (text[sz - 1] != pad) break;
478 }
479 text = text.subspan(0, sz); // adjust span in case truncated.
480 // Ensure enough room for the text, the pad byte, and the
481 // run-length encoding of the pad byte.
482 ensure_available(sz + 1 + sizeof(size_type));
483 const auto padlen{static_cast<size_type>(maxlen - sz)};
484 append_bytes(text); // append bytes to the buffer.
485 encode(static_cast<std::uint8_t>(pad)); // encode as unsigned byte
486 encode(padlen); // logical run-length of the pad byte.
487 return *this;
488 }

592 std::uint8_t u;
593 decode(u);
