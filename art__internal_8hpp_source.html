45[[nodiscard, gnu::pure]] inline int compare(const void *a, const size_t alen,
46 const void *b, const size_t blen) {
47 // TODO(thompsonbry) consider changing this over to std::span
48 // arguments and do not let the (ptr,len) pattern progagate
49 // outwards.
50 const auto shared_length = std::min(alen, blen);
51 auto ret = std::memcmp(a, b, shared_length);
52 ret = (ret != 0) ? ret : ((alen == blen) ? 0 : (alen < blen ? -1 : 1));
53 return ret;
54}

92struct [[nodiscard]] basic_art_key final {
93 private:
96 [[nodiscard, gnu::const]] static UNODB_DETAIL_CONSTEXPR_NOT_MSVC KeyType
97 make_binary_comparable(KeyType k) noexcept {
98#ifdef UNODB_DETAIL_LITTLE_ENDIAN
99 return bswap(k);
100#else
101#error Needs implementing
102#endif
103 }
104
105 public:
111 template <typename U = KeyType,
112 typename std::enable_if<std::is_integral<U>::value, int>::type = 0>
113 UNODB_DETAIL_CONSTEXPR_NOT_MSVC explicit basic_art_key(KeyType key_) noexcept
114 : key{make_binary_comparable(key_)} {}
115
118 UNODB_DETAIL_CONSTEXPR_NOT_MSVC explicit basic_art_key(key_view key_) noexcept
119 : key{key_} {}
120
122 [[nodiscard, gnu::pure]] constexpr int cmp(
123 basic_art_key<KeyType> key2) const noexcept {
124 if constexpr (std::is_same_v<KeyType, key_view>) {
125 return compare(&key, sizeof(KeyType), &key2.key, sizeof(KeyType));
126 } else {
127 return std::memcmp(&key, &key2.key, sizeof(KeyType));
128 }
129 }
130
132 [[nodiscard, gnu::pure]] constexpr int cmp(key_view key2) const noexcept {
133 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
134 // variable length keys
135 return compare(key, key2);
136 } else {
137 // fixed width keys
138 return compare(&key, sizeof(KeyType), key2.data(), key2.size_bytes());
139 }
140 }
141
144 [[nodiscard, gnu::pure]] constexpr auto operator[](
145 std::size_t index) const noexcept {
146 if constexpr (std::is_same_v<KeyType, key_view>) {
147 return key[index];
148 } else {
149 // The key_bytes[] provides access the different byte positions
150 // in the primitive type [Key key].
151 UNODB_DETAIL_ASSERT(index < sizeof(KeyType));
152 return key_bytes[index];
153 }
154 }
155
165 [[nodiscard, gnu::pure]] constexpr key_view get_key_view() const noexcept {
166 if constexpr (std::is_same_v<KeyType, key_view>) {
167 return key;
168 } else {
169 return key_view(reinterpret_cast<const std::byte *>(&key), sizeof(key));
170 }
171 }
172
177 [[nodiscard, gnu::pure]] constexpr std::uint64_t get_u64() const noexcept {
178 if constexpr (std::is_same_v<KeyType, key_view>) {
179 return unodb::detail::get_u64(key);
180 } else {
181 return static_cast<std::uint64_t>(key);
182 }
183 }
184
195 constexpr void shift_right(const std::size_t num_bytes) noexcept {
196 if constexpr (std::is_same_v<KeyType, key_view>) {
197 UNODB_DETAIL_ASSERT(num_bytes <= key.size_bytes());
198 key = key.subspan(num_bytes);
199 } else {
200 UNODB_DETAIL_ASSERT(num_bytes <= sizeof(KeyType));
201 key >>= (num_bytes * 8);
202 }
203 }
204
206 constexpr size_t size() const noexcept {
207 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
208 return key.size_bytes();
209 } else {
210 return sizeof(KeyType);
211 }
212 }
213
214 union {
218 KeyType key;
219
224 std::array<std::byte, sizeof(KeyType)> key_bytes;
225 };
226
227 static void static_asserts() {
228 static_assert(std::is_trivially_copyable_v<basic_art_key<KeyType>>);
229 static_assert(sizeof(basic_art_key<KeyType>) == sizeof(KeyType));
230 }
231

240 friend std::ostream &operator<<(std ::ostream &os, const basic_art_key &k) {
241 k.dump(os);
242 return os;
243 }
244}; // class basic_art_key

324template <class Header>
325class [[nodiscard]] basic_node_ptr {
326 public:
327 using header_type = Header;
328
329 // The default constructor does not initialize fields: it is used by
330 // std::array and we don't want to initialize to zero or any other value there
331 // at construction time.
332 // cppcheck-suppress uninitMemberVar
333 constexpr basic_node_ptr() noexcept = default;
334
335 explicit basic_node_ptr(std::nullptr_t) noexcept
336 : tagged_ptr{reinterpret_cast<std::uintptr_t>(nullptr)} {}
337
338 // construct a node pointer given a raw pointer and a node type.
339 //
340 // Note: The constructor casts away [const] for use when the
341 // node_ptr will be [const].
342 basic_node_ptr(const header_type *ptr UNODB_DETAIL_LIFETIMEBOUND,
343 unodb::node_type type) noexcept
344 : tagged_ptr{tag_ptr(const_cast<header_type *>(ptr), type)} {}
345
346 basic_node_ptr<Header> &operator=(std::nullptr_t) noexcept {
347 tagged_ptr = reinterpret_cast<std::uintptr_t>(nullptr);
348 return *this;
349 }
350
351 [[nodiscard, gnu::pure]] constexpr auto type() const noexcept {
352 return static_cast<unodb::node_type>(tagged_ptr & tag_bit_mask);
353 }
354
355 [[nodiscard, gnu::pure]] constexpr auto raw_val() const noexcept {
356 return tagged_ptr;
357 }
358
359 template <class T>
360 [[nodiscard, gnu::pure]] auto *ptr() const noexcept {
361 return reinterpret_cast<T>(tagged_ptr & ptr_bit_mask);
362 }
363
364 // same raw_val means same type and same ptr.
365 [[nodiscard, gnu::pure]] constexpr auto operator==(
366 const basic_node_ptr &other) const noexcept {
367 return tagged_ptr == other.tagged_ptr;
368 }
369
370 [[nodiscard, gnu::pure]] auto operator==(std::nullptr_t) const noexcept {
371 return tagged_ptr == reinterpret_cast<std::uintptr_t>(nullptr);
372 }
373
374 [[nodiscard, gnu::pure]] auto operator!=(std::nullptr_t) const noexcept {
375 return tagged_ptr != reinterpret_cast<std::uintptr_t>(nullptr);
376 }
377
378 private:
379 std::uintptr_t tagged_ptr;
380
381 [[nodiscard, gnu::const]] static std::uintptr_t tag_ptr(
382 Header *ptr_, unodb::node_type tag) noexcept {
383 const auto uintptr = reinterpret_cast<std::uintptr_t>(ptr_);
384 const auto result =
385 uintptr | static_cast<std::underlying_type_t<decltype(tag)>>(tag);
386 UNODB_DETAIL_ASSERT((result & ptr_bit_mask) == uintptr);
387 return result;
388 }
389
390 [[nodiscard, gnu::const]] static constexpr unsigned mask_bits_needed(
391 unsigned count) noexcept {
392 return count < 2 ? 1 : 1 + mask_bits_needed(count >> 1U);
393 }
394
395 static constexpr auto lowest_non_tag_bit =
396 1ULL << mask_bits_needed(node_type_count);
397 static constexpr auto tag_bit_mask = lowest_non_tag_bit - 1;
398 static constexpr auto ptr_bit_mask = ~tag_bit_mask;
399
400 static auto static_asserts() {
401 static_assert(sizeof(basic_node_ptr<Header>) == sizeof(void *));
402 static_assert(alignof(header_type) - 1 > lowest_non_tag_bit);
403 }
404};

407// A buffer containing an expandable binary comparable key. This is
408// used to track the key by the iterator as things are pushed and
409// popped on the stack.
410class key_buffer {
411 // TODO(thompsonbry) This data structure (std::vector but with
412 // unstable &data[0] pointer and initial stack allocation) is
413 // repeated and could be refactored out into a base class.
414 protected:
416 [[nodiscard]] size_t capacity() const noexcept { return cap; }
417
419 [[nodiscard]] size_t size_bytes() const noexcept { return off; }
420
422 void ensure_available(size_t req) {
423 if (UNODB_DETAIL_UNLIKELY(off + req > cap)) {
424 ensure_capacity(off + req); // resize
425 }
426 }
427
428 public:
432 key_buffer() noexcept = default;
433
434 ~key_buffer() {
435 if (cap > sizeof(ibuf)) { // free old buffer iff allocated
436 detail::free_aligned(buf);
437 }
438 }
439
441 void reset() { off = 0; }
442
445 [[nodiscard]] key_view get_key_view() const noexcept {
446 return key_view(buf, off);
447 }
448
450 void push(std::byte v) {
451 ensure_available(sizeof(v));
452 buf[off++] = v;
453 }
454
456 void push(key_view v) {
457 const auto n = v.size_bytes();
458 ensure_available(n);
459 std::memcpy(buf + off, v.data(), n);
460 off += n;
461 }
462
464 void pop(size_t n) {
465 UNODB_DETAIL_ASSERT(off >= n);
466 off -= n;
467 }
468
469 private:
472 void ensure_capacity(size_t min_capacity) {
473 unodb::detail::ensure_capacity(buf, cap, off, min_capacity);
474 }
475
477 std::byte ibuf[detail::INITIAL_BUFFER_CAPACITY];
478
481 std::byte *buf{&ibuf[0]};
482 size_t cap{sizeof(ibuf)}; // current buffer capacity
483 size_t off{0}; // #of bytes in the buffer having valid data.
484}; // class key_buffer
