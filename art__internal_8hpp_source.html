40// Internal ART key in binary-comparable format
41template <typename KeyType>
42struct [[nodiscard]] basic_art_key final {
43 // Convert an external key into an internal key supporting lexicographic
44 // comparison. This is only intended for key types for which simple
45 // conversions are possible. For complex keys, including multiple key
46 // components or Unicode data, the application should use a
47 // std::span<std::byte> which already supports lexicographic comparison.
48 [[nodiscard, gnu::const]] static UNODB_DETAIL_CONSTEXPR_NOT_MSVC KeyType
49 make_binary_comparable(KeyType key) noexcept;
50
51 // Convert an internal key into an external key. This is only intended for key
52 // types for which simple conversions are possible. For complex keys,
53 // including multiple key components or Unicode data, the application should
54 // use a std::span<std::byte> which already supports lexicographic comparison.
55 [[nodiscard, gnu::const]] static UNODB_DETAIL_CONSTEXPR_NOT_MSVC KeyType
56 make_external(KeyType key) noexcept;
57
58 constexpr basic_art_key() noexcept = default;
59
60 UNODB_DETAIL_CONSTEXPR_NOT_MSVC explicit basic_art_key(KeyType key_) noexcept
61 : key{make_binary_comparable(key_)} {}
62
63 [[nodiscard, gnu::pure]] constexpr bool operator==(
64 basic_art_key<KeyType> key2) const noexcept {
65 // FIXME This is wrong for variable length keys. It needs to
66 // consider no more bytes than the shorter key and if the two keys
67 // have the same prefix, then they are != if one is longer (and
68 // for == we can just compare the size as a short cut for this).
69 // Also needs unit tests for variable length keys.
70 return !std::memcmp(&key, &key2.key, size);
71 }
72
73 // @return -1, 0, or 1 if this key is LT, EQ, or GT the other key.
74 [[nodiscard, gnu::pure]] constexpr int cmp(
75 basic_art_key<KeyType> key2) const noexcept {
76 // FIXME This is wrong for variable length keys. It needs to
77 // consider no more bytes than the shorter key and if the two keys
78 // have the same prefix, then they are != if one is longer. Also
79 // needs unit tests for variable length keys.
80 return std::memcmp(&key, &key2.key, size);
81 }
82
83 [[nodiscard, gnu::pure]] constexpr auto operator[](
84 std::size_t index) const noexcept {
85 UNODB_DETAIL_ASSERT(index < size);
86 return key_bytes[index];
87 }
88
89 [[nodiscard, gnu::pure]] constexpr explicit operator KeyType()
90 const noexcept {
91 return key;
92 }
93
94 // return the decoded form of the key.
95 [[nodiscard, gnu::pure]] constexpr KeyType decode() const noexcept {
96 return make_external(key);
97 }
98
99 constexpr void shift_right(const std::size_t num_bytes) noexcept {
100 UNODB_DETAIL_ASSERT(num_bytes <= size);
101 key >>= (num_bytes * 8);
102 }
103
104 static constexpr auto size = sizeof(KeyType);
105
106 union {
107 KeyType key;
108 std::array<std::byte, size> key_bytes;
109 };
110
111 static void static_asserts() {
112 static_assert(std::is_trivially_copyable_v<basic_art_key<KeyType>>);
113 static_assert(sizeof(basic_art_key<KeyType>) == sizeof(KeyType));
114 }
115}; // class basic_art_key

196// basic_node_ptr is a tagged pointer (the tag is the node type). You
197// have to know statically the target type, then call
198// node_ptr_var.ptr<target_type *>.ptr() to get target_type.
199UNODB_DETAIL_DISABLE_MSVC_WARNING(26490)
200template <class Header>
201class [[nodiscard]] basic_node_ptr {
202 public:
203 using header_type = Header;
204
205 // The default constructor does not initialize fields: it is used by
206 // std::array and we don't want to initialize to zero or any other value there
207 // at construction time.
208 // cppcheck-suppress uninitMemberVar
209 constexpr basic_node_ptr() noexcept = default;
210
211 explicit basic_node_ptr(std::nullptr_t) noexcept
212 : tagged_ptr{reinterpret_cast<std::uintptr_t>(nullptr)} {}
213
214 // construct a node pointer given a raw pointer and a node type.
215 //
216 // Note: The constructor casts away [const] for use when the
217 // node_ptr will be [const].
218 basic_node_ptr(const header_type *ptr UNODB_DETAIL_LIFETIMEBOUND,
219 unodb::node_type type) noexcept
220 : tagged_ptr{tag_ptr(const_cast<header_type *>(ptr), type)} {}
221
222 basic_node_ptr<Header> &operator=(std::nullptr_t) noexcept {
223 tagged_ptr = reinterpret_cast<std::uintptr_t>(nullptr);
224 return *this;
225 }
226
227 [[nodiscard, gnu::pure]] constexpr auto type() const noexcept {
228 return static_cast<unodb::node_type>(tagged_ptr & tag_bit_mask);
229 }
230
231 [[nodiscard, gnu::pure]] constexpr auto raw_val() const noexcept {
232 return tagged_ptr;
233 }
234
235 template <class T>
236 [[nodiscard, gnu::pure]] auto *ptr() const noexcept {
237 return reinterpret_cast<T>(tagged_ptr & ptr_bit_mask);
238 }
239
240 // same raw_val means same type and same ptr.
241 [[nodiscard, gnu::pure]] auto operator==(
242 const basic_node_ptr &other) const noexcept {
243 return tagged_ptr == other.tagged_ptr;
244 }
245
246 [[nodiscard, gnu::pure]] auto operator==(std::nullptr_t) const noexcept {
247 return tagged_ptr == reinterpret_cast<std::uintptr_t>(nullptr);
248 }
249
250 [[nodiscard, gnu::pure]] auto operator!=(std::nullptr_t) const noexcept {
251 return tagged_ptr != reinterpret_cast<std::uintptr_t>(nullptr);
252 }
253
254 private:
255 std::uintptr_t tagged_ptr;
256
257 [[nodiscard, gnu::const]] static std::uintptr_t tag_ptr(
258 Header *ptr_, unodb::node_type tag) noexcept {
259 const auto uintptr = reinterpret_cast<std::uintptr_t>(ptr_);
260 const auto result =
261 uintptr | static_cast<std::underlying_type_t<decltype(tag)>>(tag);
262 UNODB_DETAIL_ASSERT((result & ptr_bit_mask) == uintptr);
263 return result;
264 }
265
266 [[nodiscard, gnu::const]] static constexpr unsigned mask_bits_needed(
267 unsigned count) noexcept {
268 return count < 2 ? 1 : 1 + mask_bits_needed(count >> 1U);
269 }
270
271 static constexpr auto lowest_non_tag_bit =
272 1ULL << mask_bits_needed(node_type_count);
273 static constexpr auto tag_bit_mask = lowest_non_tag_bit - 1;
274 static constexpr auto ptr_bit_mask = ~tag_bit_mask;
275
276 static auto static_asserts() {
277 static_assert(sizeof(basic_node_ptr<Header>) == sizeof(void *));
278 static_assert(alignof(header_type) - 1 > lowest_non_tag_bit);
279 }
280};

287// An object visited by the scan API. The visitor passed to the
288// caller's lambda by the scan for each index entry visited by the
289// scan.
290template <typename Iterator>
291class visitor {
292 friend class olc_db;
293 friend class db;
294
295 protected:
296 Iterator &it;
297 explicit visitor(Iterator &it_) : it(it_) {}
298
299 public:
300 // Visit the (decoded) key.
301 //
302 // Note: The lambda MUST NOT export a reference to the visited key.
303 // If you want to access the visited key outside of the scope of a
304 // single lambda invocation, then you must make a copy of the data.
305 //
306 // Note: Key decoding can be expensive and its utility is limited to
307 // simple primitive keys. In particular, key decoding is not well
308 // defined for Unicode data in keys.
309 //
310 // TODO(thompsonbry) Variable length keys: We need to define a
311 // visitor method to visit the internal key buffer without any
312 // decoding.
313 inline auto get_key() const noexcept { return it.get_key().value(); }
314
315 // Visit the value.
316 //
317 // Note: The lambda MUST NOT export a reference to the visited
318 // value. If you to access the value outside of the scope of a
319 // single lambda invocation, then you must make a copy of the data.
320 inline auto get_value() const noexcept { return it.get_val().value(); }
321}; // class visitor
