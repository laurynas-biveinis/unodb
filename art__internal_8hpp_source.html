37[[nodiscard, gnu::pure]] inline int compare(const void *a, const size_t alen,
38 const void *b,
39 const size_t blen) noexcept {
40 // TODO(thompsonbry) consider changing this over to std::span
41 // arguments and do not let the (ptr,len) pattern progagate
42 // outwards.
43 const auto shared_length = std::min(alen, blen);
44 auto ret = std::memcmp(a, b, shared_length);
45 ret = (ret != 0) ? ret : ((alen == blen) ? 0 : (alen < blen ? -1 : 1));
46 return ret;
47}

115 [[nodiscard, gnu::pure]] constexpr int cmp(
116 basic_art_key<KeyType> key2) const noexcept {
117 if constexpr (std::is_same_v<KeyType, key_view>) {
118 return compare(&key, sizeof(KeyType), &key2.key, sizeof(KeyType));
119 } else {
120 return std::memcmp(&key, &key2.key, sizeof(KeyType));
121 }
122 }
123
125 [[nodiscard, gnu::pure]] constexpr int cmp(key_view key2) const noexcept {
126 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
127 // variable length keys
128 return compare(key, key2);
129 } else {
130 // fixed width keys
131 return compare(&key, sizeof(KeyType), key2.data(), key2.size_bytes());
132 }
133 }
134
137 [[nodiscard, gnu::pure]] constexpr auto operator[](
138 std::size_t index) const noexcept {
139 if constexpr (std::is_same_v<KeyType, key_view>) {
140 return key[index];
141 } else {
142 // The key_bytes[] provides access the different byte positions
143 // in the primitive type [Key key].
144 UNODB_DETAIL_ASSERT(index < sizeof(KeyType));
145 return key_bytes[index];
146 }
147 }
148
158 [[nodiscard, gnu::pure]] constexpr key_view get_key_view() const noexcept {
159 if constexpr (std::is_same_v<KeyType, key_view>) {
160 return key;
161 } else {
162 return key_view(reinterpret_cast<const std::byte *>(&key), sizeof(key));
163 }
164 }
165
170 [[nodiscard, gnu::pure]] constexpr std::uint64_t get_u64() const noexcept {
171 if constexpr (std::is_same_v<KeyType, key_view>) {
172 return unodb::detail::get_u64(key);
173 } else {
174 return static_cast<std::uint64_t>(key);
175 }
176 }
177
188 constexpr void shift_right(const std::size_t num_bytes) noexcept {
189 if constexpr (std::is_same_v<KeyType, key_view>) {
190 UNODB_DETAIL_ASSERT(num_bytes <= key.size_bytes());
191 key = key.subspan(num_bytes);
192 } else {
193 UNODB_DETAIL_ASSERT(num_bytes <= sizeof(KeyType));
194 key >>= (num_bytes * 8);
195 }
196 }
197
199 constexpr size_t size() const noexcept {
200 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
201 return key.size_bytes();
202 } else {
203 return sizeof(KeyType);
204 }
205 }

211 KeyType key;
212
217 std::array<std::byte, sizeof(KeyType)> key_bytes;
218 };

316template <class Header>
317class [[nodiscard]] basic_node_ptr {
318 public:
319 using header_type = Header;
320
321 // The default constructor does not initialize fields: it is used by
322 // std::array and we don't want to initialize to zero or any other value there
323 // at construction time.
324 // cppcheck-suppress uninitMemberVar
325 constexpr basic_node_ptr() noexcept = default;
326
327 explicit basic_node_ptr(std::nullptr_t) noexcept
328 : tagged_ptr{reinterpret_cast<std::uintptr_t>(nullptr)} {}
329
330 // construct a node pointer given a raw pointer and a node type.
331 //
332 // Note: The constructor casts away [const] for use when the
333 // node_ptr will be [const].
334 basic_node_ptr(const header_type *ptr UNODB_DETAIL_LIFETIMEBOUND,
335 unodb::node_type type) noexcept
336 : tagged_ptr{tag_ptr(const_cast<header_type *>(ptr), type)} {}
337
338 basic_node_ptr<Header> &operator=(std::nullptr_t) noexcept {
339 tagged_ptr = reinterpret_cast<std::uintptr_t>(nullptr);
340 return *this;
341 }
342
343 [[nodiscard, gnu::pure]] constexpr auto type() const noexcept {
344 return static_cast<unodb::node_type>(tagged_ptr & tag_bit_mask);
345 }
346
347 [[nodiscard, gnu::pure]] constexpr auto raw_val() const noexcept {
348 return tagged_ptr;
349 }
350
351 template <class T>
352 [[nodiscard, gnu::pure]] auto *ptr() const noexcept {
353 return reinterpret_cast<T>(tagged_ptr & ptr_bit_mask);
354 }
355
356 // same raw_val means same type and same ptr.
357 [[nodiscard, gnu::pure]] constexpr bool operator==(
358 const basic_node_ptr &other) const noexcept {
359 return tagged_ptr == other.tagged_ptr;
360 }
361
362 [[nodiscard, gnu::pure]] bool operator==(std::nullptr_t) const noexcept {
363 return tagged_ptr == reinterpret_cast<std::uintptr_t>(nullptr);
364 }
365
366 private:
367 std::uintptr_t tagged_ptr;
368
369 [[nodiscard, gnu::const]] static std::uintptr_t tag_ptr(
370 Header *ptr_, unodb::node_type tag) noexcept {
371 const auto uintptr = reinterpret_cast<std::uintptr_t>(ptr_);
372 const auto result =
373 uintptr | static_cast<std::underlying_type_t<decltype(tag)>>(tag);
374 UNODB_DETAIL_ASSERT((result & ptr_bit_mask) == uintptr);
375 return result;
376 }
377
378 [[nodiscard, gnu::const]] static constexpr unsigned mask_bits_needed(
379 unsigned count) noexcept {
380 return count < 2 ? 1 : 1 + mask_bits_needed(count >> 1U);
381 }
382
383 static constexpr auto lowest_non_tag_bit =
384 1ULL << mask_bits_needed(node_type_count);
385 static constexpr auto tag_bit_mask = lowest_non_tag_bit - 1;
386 static constexpr auto ptr_bit_mask = ~tag_bit_mask;
387
388 static auto static_asserts() {
389 static_assert(sizeof(basic_node_ptr<Header>) == sizeof(void *));
390 static_assert(alignof(header_type) - 1 > lowest_non_tag_bit);
391 }
392};

395// A buffer containing an expandable binary comparable key. This is
396// used to track the key by the iterator as things are pushed and
397// popped on the stack.
398class key_buffer {
399 // TODO(thompsonbry) This data structure (std::vector but with
400 // unstable &data[0] pointer and initial stack allocation) is
401 // repeated and could be refactored out into a base class.
402 protected:
404 [[nodiscard]] size_t capacity() const noexcept { return cap; }
405
407 [[nodiscard]] size_t size_bytes() const noexcept { return off; }
408
410 void ensure_available(size_t req) {
411 if (UNODB_DETAIL_UNLIKELY(off + req > cap)) {
412 ensure_capacity(off + req); // resize
413 }
414 }
415
416 public:
420 key_buffer() noexcept = default;
421
422 ~key_buffer() {
423 if (cap > sizeof(ibuf)) { // free old buffer iff allocated
424 detail::free_aligned(buf);
425 }
426 }
427
429 void reset() noexcept { off = 0; }
430
433 [[nodiscard]] key_view get_key_view() const noexcept {
434 return key_view(buf, off);
435 }
436
438 void push(std::byte v) {
439 ensure_available(sizeof(v));
440 buf[off++] = v;
441 }
442
444 void push(key_view v) {
445 const auto n = v.size_bytes();
446 ensure_available(n);
447 std::memcpy(buf + off, v.data(), n);
448 off += n;
449 }
450
452 void pop(size_t n) noexcept {
453 UNODB_DETAIL_ASSERT(off >= n);
454 off -= n;
455 }
456
457 key_buffer(const key_buffer &) = delete;
458 key_buffer(key_buffer &&) = delete;
459 key_buffer &operator=(const key_buffer &) = delete;
460 key_buffer &operator=(key_buffer &&) = delete;
461
462 private:
465 void ensure_capacity(size_t min_capacity) {
466 unodb::detail::ensure_capacity(buf, cap, off, min_capacity);
467 }
468
470 std::byte ibuf[detail::INITIAL_BUFFER_CAPACITY];
471
474 std::byte *buf{&ibuf[0]};
475 size_t cap{sizeof(ibuf)}; // current buffer capacity
476 size_t off{0}; // #of bytes in the buffer having valid data.
477}; // class key_buffer
