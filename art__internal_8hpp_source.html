37[[nodiscard, gnu::pure]] inline int compare(const void *a, const size_t alen,
38 const void *b, const size_t blen) {
39 // TODO(thompsonbry) consider changing this over to std::span
40 // arguments and do not let the (ptr,len) pattern progagate
41 // outwards.
42 const auto shared_length = std::min(alen, blen);
43 auto ret = std::memcmp(a, b, shared_length);
44 ret = (ret != 0) ? ret : ((alen == blen) ? 0 : (alen < blen ? -1 : 1));
45 return ret;
46}

114 [[nodiscard, gnu::pure]] constexpr int cmp(
115 basic_art_key<KeyType> key2) const noexcept {
116 if constexpr (std::is_same_v<KeyType, key_view>) {
117 return compare(&key, sizeof(KeyType), &key2.key, sizeof(KeyType));
118 } else {
119 return std::memcmp(&key, &key2.key, sizeof(KeyType));
120 }
121 }

124 [[nodiscard, gnu::pure]] constexpr int cmp(key_view key2) const noexcept {
125 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
126 // variable length keys
127 return compare(key, key2);
128 } else {
129 // fixed width keys
130 return compare(&key, sizeof(KeyType), key2.data(), key2.size_bytes());
131 }
132 }
133
136 [[nodiscard, gnu::pure]] constexpr auto operator[](
137 std::size_t index) const noexcept {
138 if constexpr (std::is_same_v<KeyType, key_view>) {
139 return key[index];
140 } else {
141 // The key_bytes[] provides access the different byte positions
142 // in the primitive type [Key key].
143 UNODB_DETAIL_ASSERT(index < sizeof(KeyType));
144 return key_bytes[index];
145 }
146 }
147
157 [[nodiscard, gnu::pure]] constexpr key_view get_key_view() const noexcept {
158 if constexpr (std::is_same_v<KeyType, key_view>) {
159 return key;
160 } else {
161 return key_view(reinterpret_cast<const std::byte *>(&key), sizeof(key));
162 }
163 }

169 [[nodiscard, gnu::pure]] constexpr std::uint64_t get_u64() const noexcept {
170 if constexpr (std::is_same_v<KeyType, key_view>) {
171 return unodb::detail::get_u64(key);
172 } else {
173 return static_cast<std::uint64_t>(key);
174 }
175 }

187 constexpr void shift_right(const std::size_t num_bytes) noexcept {
188 if constexpr (std::is_same_v<KeyType, key_view>) {
189 UNODB_DETAIL_ASSERT(num_bytes <= key.size_bytes());
190 key = key.subspan(num_bytes);
191 } else {
192 UNODB_DETAIL_ASSERT(num_bytes <= sizeof(KeyType));
193 key >>= (num_bytes * 8);
194 }
195 }
196
198 constexpr size_t size() const noexcept {
199 if constexpr (std::is_same_v<KeyType, unodb::key_view>) {
200 return key.size_bytes();
201 } else {
202 return sizeof(KeyType);
203 }
204 }

315template <class Header>
316class [[nodiscard]] basic_node_ptr {
317 public:
318 using header_type = Header;
319
320 // The default constructor does not initialize fields: it is used by
321 // std::array and we don't want to initialize to zero or any other value there
322 // at construction time.
323 // cppcheck-suppress uninitMemberVar
324 constexpr basic_node_ptr() noexcept = default;
325
326 explicit basic_node_ptr(std::nullptr_t) noexcept
327 : tagged_ptr{reinterpret_cast<std::uintptr_t>(nullptr)} {}
328
329 // construct a node pointer given a raw pointer and a node type.
330 //
331 // Note: The constructor casts away [const] for use when the
332 // node_ptr will be [const].
333 basic_node_ptr(const header_type *ptr UNODB_DETAIL_LIFETIMEBOUND,
334 unodb::node_type type) noexcept
335 : tagged_ptr{tag_ptr(const_cast<header_type *>(ptr), type)} {}
336
337 basic_node_ptr<Header> &operator=(std::nullptr_t) noexcept {
338 tagged_ptr = reinterpret_cast<std::uintptr_t>(nullptr);
339 return *this;
340 }
341
342 [[nodiscard, gnu::pure]] constexpr auto type() const noexcept {
343 return static_cast<unodb::node_type>(tagged_ptr & tag_bit_mask);
344 }
345
346 [[nodiscard, gnu::pure]] constexpr auto raw_val() const noexcept {
347 return tagged_ptr;
348 }
349
350 template <class T>
351 [[nodiscard, gnu::pure]] auto *ptr() const noexcept {
352 return reinterpret_cast<T>(tagged_ptr & ptr_bit_mask);
353 }
354
355 // same raw_val means same type and same ptr.
356 [[nodiscard, gnu::pure]] constexpr auto operator==(
357 const basic_node_ptr &other) const noexcept {
358 return tagged_ptr == other.tagged_ptr;
359 }
360
361 [[nodiscard, gnu::pure]] auto operator==(std::nullptr_t) const noexcept {
362 return tagged_ptr == reinterpret_cast<std::uintptr_t>(nullptr);
363 }
364
365 [[nodiscard, gnu::pure]] auto operator!=(std::nullptr_t) const noexcept {
366 return tagged_ptr != reinterpret_cast<std::uintptr_t>(nullptr);
367 }
368
369 private:
370 std::uintptr_t tagged_ptr;
371
372 [[nodiscard, gnu::const]] static std::uintptr_t tag_ptr(
373 Header *ptr_, unodb::node_type tag) noexcept {
374 const auto uintptr = reinterpret_cast<std::uintptr_t>(ptr_);
375 const auto result =
376 uintptr | static_cast<std::underlying_type_t<decltype(tag)>>(tag);
377 UNODB_DETAIL_ASSERT((result & ptr_bit_mask) == uintptr);
378 return result;
379 }
380
381 [[nodiscard, gnu::const]] static constexpr unsigned mask_bits_needed(
382 unsigned count) noexcept {
383 return count < 2 ? 1 : 1 + mask_bits_needed(count >> 1U);
384 }
385
386 static constexpr auto lowest_non_tag_bit =
387 1ULL << mask_bits_needed(node_type_count);
388 static constexpr auto tag_bit_mask = lowest_non_tag_bit - 1;
389 static constexpr auto ptr_bit_mask = ~tag_bit_mask;
390
391 static auto static_asserts() {
392 static_assert(sizeof(basic_node_ptr<Header>) == sizeof(void *));
393 static_assert(alignof(header_type) - 1 > lowest_non_tag_bit);
394 }
395};

398// A buffer containing an expandable binary comparable key. This is
399// used to track the key by the iterator as things are pushed and
400// popped on the stack.
401class key_buffer {
402 // TODO(thompsonbry) This data structure (std::vector but with
403 // unstable &data[0] pointer and initial stack allocation) is
404 // repeated and could be refactored out into a base class.
405 protected:
407 [[nodiscard]] size_t capacity() const noexcept { return cap; }
408
410 [[nodiscard]] size_t size_bytes() const noexcept { return off; }
411
413 void ensure_available(size_t req) {
414 if (UNODB_DETAIL_UNLIKELY(off + req > cap)) {
415 ensure_capacity(off + req); // resize
416 }
417 }
418
419 public:
423 key_buffer() noexcept = default;
424
425 ~key_buffer() {
426 if (cap > sizeof(ibuf)) { // free old buffer iff allocated
427 detail::free_aligned(buf);
428 }
429 }
430
432 void reset() { off = 0; }
433
436 [[nodiscard]] key_view get_key_view() const noexcept {
437 return key_view(buf, off);
438 }
439
441 void push(std::byte v) {
442 ensure_available(sizeof(v));
443 buf[off++] = v;
444 }
445
447 void push(key_view v) {
448 const auto n = v.size_bytes();
449 ensure_available(n);
450 std::memcpy(buf + off, v.data(), n);
451 off += n;
452 }
453
455 void pop(size_t n) {
456 UNODB_DETAIL_ASSERT(off >= n);
457 off -= n;
458 }
459
460 private:
463 void ensure_capacity(size_t min_capacity) {
464 unodb::detail::ensure_capacity(buf, cap, off, min_capacity);
465 }
466
468 std::byte ibuf[detail::INITIAL_BUFFER_CAPACITY];
469
472 std::byte *buf{&ibuf[0]};
473 size_t cap{sizeof(ibuf)}; // current buffer capacity
474 size_t off{0}; // #of bytes in the buffer having valid data.
475}; // class key_buffer
