45[[noreturn, gnu::cold]] UNODB_DETAIL_HEADER_NOINLINE void msg_stacktrace_abort(
46 std::string_view msg) noexcept {
47 UNODB_DETAIL_FAIL_ON_NTH_ALLOCATION(0);
48 std::ostringstream buf;
49 buf << msg;
50#ifdef UNODB_DETAIL_BOOST_STACKTRACE
51 buf << boost::stacktrace::stacktrace();
52#else
53 buf << "(stacktrace not available, not compiled with Boost.Stacktrace)\n";
54#endif
55 std::cerr << buf.str();
56 std::abort();
