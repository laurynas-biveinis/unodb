68[[noreturn, gnu::cold]] UNODB_DETAIL_HEADER_NOINLINE void msg_stacktrace_abort(
69 std::string_view msg) noexcept {
70 UNODB_DETAIL_FAIL_ON_NTH_ALLOCATION(0);
71 std::ostringstream buf;
72 buf << msg;
73#ifdef UNODB_DETAIL_BOOST_STACKTRACE
74 buf << boost::stacktrace::stacktrace();
75#else
76 buf << "(stacktrace not available, not compiled with Boost.Stacktrace)\n";
77#endif
78 std::cerr << buf.str();
79 std::abort();
