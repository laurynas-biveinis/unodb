78template <class Db>
79void assert_value_eq(const typename Db::get_result &result,
80 unodb::value_view expected) noexcept {
81 if constexpr (std::is_same_v<Db, unodb::mutex_db<typename Db::key_type>>) {
82 UNODB_DETAIL_ASSERT(result.second.owns_lock());
83 UNODB_DETAIL_ASSERT(result.first.has_value());
84 UNODB_ASSERT_TRUE(std::ranges::equal(*result.first, expected));
85 } else {
86 UNODB_DETAIL_ASSERT(result.has_value());
87 UNODB_ASSERT_TRUE(std::ranges::equal(*result, expected));
88 }

97 unodb::detail::dump_key(msg, key);
98 const testing::ScopedTrace trace(file, line, msg.str());

103 unodb::detail::dump_key(std::cerr, key);

107 // LCOV_EXCL_STOP

252 void insert(key_type k, unodb::value_view v, bool bypass_verifier = false) {
253 const auto empty_before = test_db.empty();
254#ifdef UNODB_DETAIL_WITH_STATS
255 const auto mem_use_before =
256 parallel_test ? 0 : test_db.get_current_memory_use();
257 const auto node_counts_before = test_db.get_node_counts();
258 const auto growing_inodes_before = test_db.get_growing_inode_counts();
259 const auto shrinking_inodes_before = test_db.get_shrinking_inode_counts();
260 const auto key_prefix_splits_before = test_db.get_key_prefix_splits();
261#endif // UNODB_DETAIL_WITH_STATS
262
263 try {
264 do_insert(k, v);
265 } catch (...) {
266 if (!parallel_test) {
267 UNODB_ASSERT_EQ(empty_before, test_db.empty());
268#ifdef UNODB_DETAIL_WITH_STATS
269 UNODB_ASSERT_EQ(mem_use_before, test_db.get_current_memory_use());
270 UNODB_ASSERT_THAT(test_db.get_node_counts(),
271 ::testing::ElementsAreArray(node_counts_before));
272 UNODB_ASSERT_THAT(test_db.get_growing_inode_counts(),
273 ::testing::ElementsAreArray(growing_inodes_before));
274 UNODB_ASSERT_THAT(test_db.get_shrinking_inode_counts(),
275 ::testing::ElementsAreArray(shrinking_inodes_before));
276 UNODB_ASSERT_EQ(test_db.get_key_prefix_splits(),
277 key_prefix_splits_before);
278#endif // UNODB_DETAIL_WITH_STATS
279 }
280 throw;
281 }
282
283 UNODB_ASSERT_FALSE(test_db.empty());
284
285#ifdef UNODB_DETAIL_WITH_STATS
286 const auto mem_use_after = test_db.get_current_memory_use();
287 if (parallel_test)
288 UNODB_ASSERT_GT(mem_use_after, 0);
289 else
290 UNODB_ASSERT_LT(mem_use_before, mem_use_after);
291
292 const auto leaf_count_after =
293 test_db.template get_node_count<unodb::node_type::LEAF>();
294 if (parallel_test)
295 UNODB_ASSERT_GT(leaf_count_after, 0);
296 else
297 UNODB_ASSERT_EQ(leaf_count_after,
298 node_counts_before[as_i<unodb::node_type::LEAF>] + 1);
299#endif // UNODB_DETAIL_WITH_STATS
300
301 if (!bypass_verifier) {
302#ifndef NDEBUG
303 allocation_failure_injector::reset();
304#endif
305 const auto [pos, insert_succeeded] = values.try_emplace(k, v);
306 (void)pos;
307 UNODB_ASSERT_TRUE(insert_succeeded);
308 }
309 }
