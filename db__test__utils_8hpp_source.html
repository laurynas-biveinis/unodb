77UNODB_DETAIL_DISABLE_MSVC_WARNING(6326)
78template <class Db>
79void assert_value_eq(const typename Db::get_result &result,
80 unodb::value_view expected) noexcept {
81 if constexpr (std::is_same_v<Db, unodb::mutex_db>) {
82 UNODB_DETAIL_ASSERT(result.second.owns_lock());
83 UNODB_DETAIL_ASSERT(result.first.has_value());
84 UNODB_ASSERT_TRUE(std::equal(std::cbegin(*result.first),
85 std::cend(*result.first),
86 std::cbegin(expected), std::cend(expected)));
87 } else {
88 UNODB_DETAIL_ASSERT(result.has_value());
89 UNODB_ASSERT_TRUE(std::equal(std::cbegin(*result), std::cend(*result),
90 std::cbegin(expected), std::cend(expected)));
91 }
92}

95template <class Db>
96void do_assert_result_eq(const Db &db, unodb::key key,
97 unodb::value_view expected, const char *file,
98 int line) {
99 std::ostringstream msg;
100 unodb::detail::dump_key(msg, key);
101 const testing::ScopedTrace trace(file, line, msg.str());
102 const auto result = db.get(key);
103 if (!Db::key_found(result)) {
104 // LCOV_EXCL_START
105 std::cerr << "db.get did not find ";
106 unodb::detail::dump_key(std::cerr, key);
107 std::cerr << '\n';
108 db.dump(std::cerr);
109 FAIL();
110 // LCOV_EXCL_STOP
111 }
112 assert_value_eq<Db>(result, expected);
113}

224 UNODB_DETAIL_DISABLE_MSVC_WARNING(6326)
225 void insert(unodb::key k, unodb::value_view v, bool bypass_verifier = false) {
226 const auto empty_before = test_db.empty();
227#ifdef UNODB_DETAIL_WITH_STATS
228 const auto mem_use_before =
229 parallel_test ? 0 : test_db.get_current_memory_use();
230 const auto node_counts_before = test_db.get_node_counts();
231 const auto growing_inodes_before = test_db.get_growing_inode_counts();
232 const auto shrinking_inodes_before = test_db.get_shrinking_inode_counts();
233 const auto key_prefix_splits_before = test_db.get_key_prefix_splits();
234#endif // UNODB_DETAIL_WITH_STATS
235
236 try {
237 do_insert(k, v);
238 } catch (...) {
239 if (!parallel_test) {
240 UNODB_ASSERT_EQ(empty_before, test_db.empty());
241#ifdef UNODB_DETAIL_WITH_STATS
242 UNODB_ASSERT_EQ(mem_use_before, test_db.get_current_memory_use());
243 UNODB_ASSERT_THAT(test_db.get_node_counts(),
244 ::testing::ElementsAreArray(node_counts_before));
245 UNODB_ASSERT_THAT(test_db.get_growing_inode_counts(),
246 ::testing::ElementsAreArray(growing_inodes_before));
247 UNODB_ASSERT_THAT(test_db.get_shrinking_inode_counts(),
248 ::testing::ElementsAreArray(shrinking_inodes_before));
249 UNODB_ASSERT_EQ(test_db.get_key_prefix_splits(),
250 key_prefix_splits_before);
251#endif // UNODB_DETAIL_WITH_STATS
252 }
253 throw;
254 }
255
256 UNODB_ASSERT_FALSE(test_db.empty());
257
258#ifdef UNODB_DETAIL_WITH_STATS
259 const auto mem_use_after = test_db.get_current_memory_use();
260 if (parallel_test)
261 UNODB_ASSERT_GT(mem_use_after, 0);
262 else
263 UNODB_ASSERT_LT(mem_use_before, mem_use_after);
264
265 const auto leaf_count_after =
266 test_db.template get_node_count<unodb::node_type::LEAF>();
267 if (parallel_test)
268 UNODB_ASSERT_GT(leaf_count_after, 0);
269 else
270 UNODB_ASSERT_EQ(leaf_count_after,
271 node_counts_before[as_i<unodb::node_type::LEAF>] + 1);
272#endif // UNODB_DETAIL_WITH_STATS
273
274 if (!bypass_verifier) {
275#ifndef NDEBUG
276 allocation_failure_injector::reset();
277#endif
278 const auto [pos, insert_succeeded] = values.try_emplace(k, v);
279 (void)pos;
280 UNODB_ASSERT_TRUE(insert_succeeded);
281 }
282 }
