191 [[nodiscard]] ikey_type<Db> to_ikey(typename Db::key_type key) const {
192 if constexpr (std::is_same_v<key_type, unodb::key_view>) {
193 // Allocate a vector, make a copy of the key into the vector,
194 // and return a shared_ptr to that vector.
195 UNODB_DETAIL_PAUSE_HEAP_TRACKING_GUARD()
196 const auto nbytes = key.size_bytes();
197 auto *vec = new std::vector<std::byte>(nbytes);
198 std::memcpy(vec->data(), key.data(), nbytes);
199 return key_wrapper{vec};
200 } else {
201 return key; // NOP
202 }
203 }

205 template <typename T>
206 key_type coerce_key_internal(T key) {
207 if constexpr (std::is_same_v<key_type, unodb::key_view>) { // Db<key_view>?
208 if constexpr (std::is_same_v<T, unodb::key_view>) { // Given key_view?
209 // key_view pass through
210 return key;
211 } else {
212 // type promote and then encode the key into a key_view.
213 return make_key(static_cast<u64>(key));
214 }
215 } else { // type promotion.
216 return static_cast<u64>(key);
217 }
218 }

250 unodb::key_view make_key(std::uint64_t k) {
251 constexpr auto sz{sizeof(k)};
252 UNODB_DETAIL_PAUSE_HEAP_TRACKING_GUARD()
253 // Encode the key, emplace an array into the list of encoded keys
254 // that we are tracking, and copy the encoded key into that
255 // emplaced array.
256 unodb::key_encoder enc;
257 auto kv{enc.encode(k).get_key_view()};
258 key_views.emplace_back(std::array<std::byte, sz>{});
259 auto &a = key_views.back(); // a *reference* to data emplaced_back.
260 std::copy(kv.data(), kv.data() + sz, a.begin()); // copy data into array.
261 // Return a key_view backed by the array that we just put on that
262 // list.
263 return unodb::key_view(a.data(), sz); // view of array's data.
264 }

370 void insert_internal(key_type k, unodb::value_view v,
371 bool bypass_verifier = false) {
372 const auto empty_before = test_db.empty();
373#ifdef UNODB_DETAIL_WITH_STATS
374 const auto mem_use_before =
375 parallel_test ? 0 : test_db.get_current_memory_use();
376 const auto node_counts_before = test_db.get_node_counts();
377 const auto growing_inodes_before = test_db.get_growing_inode_counts();
378 const auto shrinking_inodes_before = test_db.get_shrinking_inode_counts();
379 const auto key_prefix_splits_before = test_db.get_key_prefix_splits();
380#endif // UNODB_DETAIL_WITH_STATS
381
382 try {
383 do_insert(k, v);
384 } catch (...) {
385 if (!parallel_test) {
386 UNODB_ASSERT_EQ(empty_before, test_db.empty());
387#ifdef UNODB_DETAIL_WITH_STATS
388 UNODB_ASSERT_EQ(mem_use_before, test_db.get_current_memory_use());
389 UNODB_ASSERT_THAT(test_db.get_node_counts(),
390 ::testing::ElementsAreArray(node_counts_before));
391 UNODB_ASSERT_THAT(test_db.get_growing_inode_counts(),
392 ::testing::ElementsAreArray(growing_inodes_before));
393 UNODB_ASSERT_THAT(test_db.get_shrinking_inode_counts(),
394 ::testing::ElementsAreArray(shrinking_inodes_before));
395 UNODB_ASSERT_EQ(test_db.get_key_prefix_splits(),
396 key_prefix_splits_before);
397#endif // UNODB_DETAIL_WITH_STATS
398 }
399 throw;
400 }
401
402 UNODB_ASSERT_FALSE(test_db.empty());
403
404#ifdef UNODB_DETAIL_WITH_STATS
405 const auto mem_use_after = test_db.get_current_memory_use();
406 if (parallel_test)
407 UNODB_ASSERT_GT(mem_use_after, 0);
408 else
409 UNODB_ASSERT_LT(mem_use_before, mem_use_after);
410
411 const auto leaf_count_after =
412 test_db.template get_node_count<unodb::node_type::LEAF>();
413 if (parallel_test)
414 UNODB_ASSERT_GT(leaf_count_after, 0);
415 else
416 UNODB_ASSERT_EQ(leaf_count_after,
417 node_counts_before[as_i<unodb::node_type::LEAF>] + 1);
418#endif // UNODB_DETAIL_WITH_STATS

420 if (!bypass_verifier) {
421#ifndef NDEBUG
422 allocation_failure_injector::reset();
423#endif
424 UNODB_DETAIL_PAUSE_HEAP_TRACKING_GUARD()
425 const auto [pos, insert_succeeded] = values.try_emplace(to_ikey(k), v);
426 (void)pos;
427 UNODB_ASSERT_TRUE(insert_succeeded);
428 }

432 void insert_key_range_internal(key_type start_key, std::size_t count,
433 bool bypass_verifier = false) {
434 if constexpr (std::is_same_v<key_type, key_view>) {
435 // decode to figure out the start key for the loop.
436 unodb::key_decoder dec(start_key);
437 std::uint64_t start_key_dec;
438 dec.decode(start_key_dec);
439 unodb::key_encoder enc;
440 for (auto key = start_key_dec; key < start_key_dec + count; ++key) {
441 insert(enc.reset().encode(key).get_key_view(),
442 test_values[key % test_values.size()], bypass_verifier);
443 }
444 } else {
445 for (auto key = start_key; key < start_key + count; ++key) {
446 insert(key, test_values[key % test_values.size()], bypass_verifier);
447 }
448 }
449 }
