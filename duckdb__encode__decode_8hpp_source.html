45template <typename T>
46[[nodiscard]] constexpr T msb();
47template <>
48constexpr std::uint64_t msb<std::uint64_t>() {
49 return 1ULL << 63;
50}
51template <>
52constexpr std::uint32_t msb<std::uint32_t>() {
53 return 1U << 31;

56// This method is derived from DuckDB.
57template <typename U, typename F>
58[[nodiscard]] U encode_floating_point(F x) noexcept {
59 constexpr auto msb0 = unodb::detail::msb<U>();
60 if (std::isnan(x)) {
61 return std::numeric_limits<U>::max(); // NaN
62 }
63 if (std::isinf(x)) { // (+|-) inf
64 return (x > 0) ? std::numeric_limits<U>::max() - 1 // +inf
65 : 0; // -inf
66 }
67 U buff = unodb::detail::bit_cast<U, F>(x);
68 if ((buff & msb0) == 0) {
69 buff |= msb0;
70 } else {
71 buff = ~buff;
72 }
73 return buff;
74}

76// This method is derived from DuckDB.
77template <typename F, typename U>
78[[nodiscard]] F decode_floating_point(U input) noexcept {
79 constexpr auto msb0 = unodb::detail::msb<U>();
80 if (input == std::numeric_limits<U>::max()) {
81 return std::numeric_limits<F>::quiet_NaN(); // NaN
82 }
83 if (input == std::numeric_limits<U>::max() - 1) {
84 return std::numeric_limits<F>::infinity(); // +inf
85 }
86 if (input == 0) {
87 return -std::numeric_limits<F>::infinity(); // -inf
88 }
89 if (input & msb0) {
90 input = input ^ msb0; // positive numbers - flip sign bit
91 } else {
92 input = ~input; // negative numbers - invert
93 }
94 return unodb::detail::bit_cast<F, U>(input);
95}
