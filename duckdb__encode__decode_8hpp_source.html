43template <typename T>
44[[nodiscard, gnu::const]] consteval T msb() noexcept {
45 static_assert(std::is_same_v<std::uint32_t, T> ||
46 std::is_same_v<std::uint64_t, T>);
47 if (std::is_same_v<std::uint32_t, T>) {
48 return 1U << 31U;
49 }
50 if (std::is_same_v<std::uint64_t, T>) {
51 return static_cast<T>(1ULL << 63U);
52 }
53}

73template <typename U, typename F>
74[[nodiscard]] U encode_floating_point(F x) noexcept {
75 constexpr auto msb0 = msb<U>();
76 if (std::isnan(x)) {
77 return std::numeric_limits<U>::max(); // NaN
78 }
79 if (std::isinf(x)) { // (+|-) inf
80 return (x > 0) ? std::numeric_limits<U>::max() - 1 // +inf
81 : 0; // -inf
82 }
83 auto buff = bit_cast<U, F>(x);
84 if ((buff & msb0) == 0) { // +0 and positive numbers
85 buff |= msb0;
86 } else { // negative numbers
87 buff = ~buff; // complement 1
88 }
89 return buff;
90}

100template <typename F, typename U>
101[[nodiscard]] F decode_floating_point(U input) noexcept {
102 constexpr auto msb0 = msb<U>();
103 if (input == std::numeric_limits<U>::max()) {
104 return std::numeric_limits<F>::quiet_NaN(); // NaN
105 }
106 if (input == std::numeric_limits<U>::max() - 1) {
107 return std::numeric_limits<F>::infinity(); // +inf
108 }
109 if (input == 0) {
110 return -std::numeric_limits<F>::infinity(); // -inf
111 }
112 if (input & msb0) {
113 input = input ^ msb0; // positive numbers - flip sign bit
114 } else {
115 input = ~input; // negative numbers - invert
116 }
117 return bit_cast<F, U>(input);
118}
