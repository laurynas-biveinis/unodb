37[[nodiscard]] inline void* allocate_aligned(
38 std::size_t size,
39 std::size_t alignment = __STDCPP_DEFAULT_NEW_ALIGNMENT__) {
40#ifndef NDEBUG
41 unodb::test::allocation_failure_injector::maybe_fail();
42#endif
43
44 void* result;
45
46#ifndef _MSC_VER
47 const auto err = posix_memalign(&result, alignment, size);
48 if (UNODB_DETAIL_UNLIKELY(err != 0)) result = nullptr;
49#else
50 result = _aligned_malloc(size, alignment);
51#ifndef NDEBUG
52 const auto err = UNODB_DETAIL_LIKELY(result != nullptr) ? 0 : errno;
53#endif
54#endif
55
56 UNODB_DETAIL_ASSERT(err != EINVAL);
57 // NOLINTNEXTLINE(readability-simplify-boolean-expr)
58 UNODB_DETAIL_ASSERT(result != nullptr || err == ENOMEM);
59
60 if (UNODB_DETAIL_UNLIKELY(result == nullptr)) {
61 throw std::bad_alloc{}; // LCOV_EXCL_LINE
62 }
63
64 return result;
65}
