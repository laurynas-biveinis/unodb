30[[nodiscard]] inline void* allocate_aligned(
31 std::size_t size,
32 std::size_t alignment = __STDCPP_DEFAULT_NEW_ALIGNMENT__) {
33#ifndef NDEBUG
34 unodb::test::allocation_failure_injector::maybe_fail();
35#endif
36
37 void* result;
38
39#ifndef _MSC_VER
40 const auto err = posix_memalign(&result, alignment, size);
41 if (UNODB_DETAIL_UNLIKELY(err != 0)) result = nullptr;
42#else
43 result = _aligned_malloc(size, alignment);
44#ifndef NDEBUG
45 const auto err = UNODB_DETAIL_LIKELY(result != nullptr) ? 0 : errno;
46#endif
47#endif
48
49 UNODB_DETAIL_ASSERT(err != EINVAL);
50 // NOLINTNEXTLINE(readability-simplify-boolean-expr)
51 UNODB_DETAIL_ASSERT(result != nullptr || err == ENOMEM);
52
53 if (UNODB_DETAIL_UNLIKELY(result == nullptr)) {
54 throw std::bad_alloc{}; // LCOV_EXCL_LINE
55 }
56
57 return result;
58}
