26// Something small for CI quick checks
27static constexpr auto small_concurrent_tree_size = 70000;
28// Do not OOM on a 16GB Linux test server
29static constexpr auto large_concurrent_tree_size = 2000000;

31constexpr void concurrency_ranges(::benchmark::internal::Benchmark *b,
32 int max_concurrency) {
33 for (auto i = 1; i <= max_concurrency; i *= 2)
34 b->Args({i, small_concurrent_tree_size});
35 for (auto i = 1; i <= max_concurrency; i *= 2)
36 b->Args({i, large_concurrent_tree_size});
37}

73 void parallel_get(::benchmark::State &state) {
74 const auto num_of_threads = static_cast<std::size_t>(state.range(0));
75 const auto tree_size = static_cast<std::uint64_t>(state.range(1));
76
77 test_db = std::make_unique<Db>();
78
79 for (std::uint64_t i = 0; i < tree_size; ++i)
80 insert_key(*test_db, i, values[i % values.size()]);
81
82 for (const auto _ : state) {
83 state.PauseTiming();
84 do_parallel_test(*test_db, num_of_threads, tree_size, parallel_get_worker,
85 state);
86 state.ResumeTiming();
87 }
88
89 state.SetItemsProcessed(state.range(1));
90 test_db.reset(nullptr);
91 }

140 private:
141 template <typename Worker>
142 void do_parallel_test(Db &instance, std::size_t num_of_threads,
143 std::size_t tree_size, Worker worker,
144 ::benchmark::State &state) {
145 setup();
146
147 std::vector<Thread> threads{num_of_threads - 1};
148 const std::uint64_t length{tree_size / num_of_threads};
149
150 state.ResumeTiming();
151
152 for (std::size_t i = 1; i < num_of_threads; ++i) {
153 const std::uint64_t start = i * length;
154 threads[i - 1] = Thread{worker, std::ref(instance), start, length};
155 }
156
157 worker(instance, 0, length);
158
159 for (std::size_t i = 1; i < num_of_threads; ++i) {
160 threads[i - 1].join();
161 }
162
163 end_workload_in_main_thread();
164
165 state.PauseTiming();
166
167 teardown();
168 }
169
170 static void parallel_get_worker(const Db &test_db, std::uint64_t start,
171 std::uint64_t length) {
172 for (std::uint64_t i = start; i < start + length; ++i)
173 get_existing_key(test_db, i);
174 }
175
176 static void parallel_insert_worker(Db &test_db, std::uint64_t start,
177 std::uint64_t length) {
178 for (std::uint64_t i = start; i < start + length; ++i)
179 insert_key(test_db, i, values[i % values.size()]);
180 }
181
182 static void parallel_delete_worker(Db &test_db, std::uint64_t start,
183 std::uint64_t length) {
184 for (std::uint64_t i = start; i < start + length; ++i)
185 delete_key(test_db, i);
186 }
187
188 std::unique_ptr<Db> test_db;
189};
