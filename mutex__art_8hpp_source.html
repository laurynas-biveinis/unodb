47 [[nodiscard]] auto get_internal(art_key_type k) const noexcept {
48 std::unique_lock guard{mutex};
49 const auto db_get_result{db_.get_internal(k)};
50 if (!db_get_result) {
51 guard.unlock();
52 return std::make_pair(db_get_result, std::unique_lock<std::mutex>{});
53 }
54 return std::make_pair(db_get_result, std::move(guard));
55 }

248 // Releases the mutex in the case key was not found, keeps it locked
249 // otherwise.
250 [[nodiscard]] static auto key_found(const get_result &result) noexcept {
251#ifndef NDEBUG
252 const auto &lock{result.second};
253 // NOLINTNEXTLINE(readability-simplify-boolean-expr)
254 assert(!result.first || lock.owns_lock());
255#endif
256
257 return static_cast<bool>(result.first);
258 }
