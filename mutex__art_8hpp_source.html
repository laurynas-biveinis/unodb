36 // Querying
37 [[nodiscard]] auto get(key k) const {
38 std::unique_lock guard{mutex};
39 const auto db_get_result{db_.get(k)};
40 if (!db_get_result) {
41 guard.unlock();
42 return std::make_pair(db_get_result, std::unique_lock<std::mutex>{});
43 }
44 return std::make_pair(db_get_result, std::move(guard));
45 }

75 // Scan the tree, applying the caller's lambda to each visited leaf.
76 // The tree remains locked for the duration of the scan.
77 //
78 // @param fn A function f(unodb::visitor<unodb::mutex_db::iterator>&)
79 // returning [bool::halt]. The traversal will halt if the function
80 // returns [true].
81 //
82 // @param fwd When [true] perform a forward scan, otherwise perform
83 // a reverse scan.
84 template <typename FN>
85 inline void scan(FN fn, bool fwd = true) noexcept {
86 const std::lock_guard guard{mutex};
87 db_.scan(fn, fwd);
88 }
89
90 // Scan in the indicated direction, applying the caller's lambda to
91 // each visited leaf. The tree remains locked for the duration of
92 // the scan.
93 //
94 // @param fromKey is an inclusive lower bound for the starting point
95 // of the scan.
96 //
97 // @param fn A function f(unodb::visitor<unodb::mutex_db::iterator>&)
98 // returning [bool::halt]. The traversal will halt if the function
99 // returns [true].
100 //
101 // @param fwd When [true] perform a forward scan, otherwise perform
102 // a reverse scan.
103 template <typename FN>
104 inline void scan_from(const key fromKey, FN fn, bool fwd = true) noexcept {
105 const std::lock_guard guard{mutex};
106 db_.scan_from(fromKey, fn, fwd);
107 }
108
109 // Scan a half-open key range, applying the caller's lambda to each
110 // visited leaf. The tree remains locked for the duration of the
111 // scan. The scan will proceed in lexicographic order iff fromKey
112 // is less than toKey and in reverse lexicographic order iff toKey
113 // is less than fromKey. When fromKey < toKey, the scan will visit
114 // all index entries in the half-open range [fromKey,toKey) in
115 // forward order. Otherwise the scan will visit all index entries
116 // in the half-open range (fromKey,toKey] in reverse order.
117 //
118 // @param fromKey is an inclusive bound for the starting point of
119 // the scan.
120 //
121 // @param toKey is an exclusive bound for the ending point of the
122 // scan.
123 //
124 // @param fn A function f(unodb::visitor<unodb::mutex_db::iterator>&)
125 // returning [bool::halt]. The traversal will halt if the function
126 // returns [true].
127 template <typename FN>
128 inline void scan_range(const key fromKey, const key toKey, FN fn) noexcept {
129 const std::lock_guard guard{mutex};
130 db_.scan_range(fromKey, toKey, fn);
131 }

190 // Releases the mutex in the case key was not found, keeps it locked
191 // otherwise.
192 [[nodiscard]] static auto key_found(const get_result &result) noexcept {
193#ifndef NDEBUG
194 const auto &lock{result.second};
195 // NOLINTNEXTLINE(readability-simplify-boolean-expr)
196 assert(!result.first || lock.owns_lock());
197#endif
198
199 return static_cast<bool>(result.first);
200 }
