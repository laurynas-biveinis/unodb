39 // If the search key was found, that is, the first pair member has a value,
40 // then the second member is a locked tree mutex which must be released ASAP
41 // after reading the first pair member. Otherwise, the second member is
42 // undefined.

50 [[nodiscard]] auto get_internal(art_key_type k) const {
51 std::unique_lock guard{mutex};
52 const auto db_get_result{db_.get_internal(k)};
53 if (!db_get_result) {
54 guard.unlock();
55 return std::make_pair(db_get_result, std::unique_lock<std::mutex>{});
56 }
57 return std::make_pair(db_get_result, std::move(guard));
58 }
59

252 // Releases the mutex in the case key was not found, keeps it locked
253 // otherwise.
254 [[nodiscard]] static auto key_found(const get_result &result) noexcept {
255#ifndef NDEBUG
256 const auto &lock{result.second};
257 // NOLINTNEXTLINE(readability-simplify-boolean-expr)
258 assert(!result.first || lock.owns_lock());
259#endif
260
261 return static_cast<bool>(result.first);
262 }
