45 [[nodiscard]] auto get_internal(art_key_type k) const noexcept {
46 std::unique_lock guard{mutex};
47 const auto db_get_result{db_.get_internal(k)};
48 if (!db_get_result) {
49 guard.unlock();
50 return std::make_pair(db_get_result, std::unique_lock<std::mutex>{});
51 }
52 return std::make_pair(db_get_result, std::move(guard));
53 }

249#ifndef NDEBUG
