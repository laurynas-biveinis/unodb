378 [[nodiscard]] bool cas_acquire(version_type expected,
379 version_type new_val) noexcept {
380 auto expected_val = expected.get();

389 void write_unlock() noexcept {
390 // This thread has written the previous lock word value, and no other
391 // thread may write it before the unlock, thus we can read it without
392 // ordering.
393 const auto old_lock_word = load_relaxed();
394 UNODB_DETAIL_ASSERT(old_lock_word.is_write_locked());
395
396 const auto new_lock_word = old_lock_word.get() + 2;
397 version.store(new_lock_word, std::memory_order_release);
398 }

404 void write_unlock_and_obsolete() noexcept {
405#ifndef NDEBUG

407 UNODB_DETAIL_ASSERT(!old_lock_word.is_obsolete());
408 UNODB_DETAIL_ASSERT(old_lock_word.is_write_locked());
409#endif
410
411 version.store(version_type::obsolete_lock_word,
412 std::memory_order_release);
413
414 UNODB_DETAIL_ASSERT(load_relaxed().is_obsolete());
415 }

684 [[nodiscard]] read_critical_section try_read_lock() noexcept {
685 while (true) {
686 const auto current_version = version.load_acquire();
687 if (UNODB_DETAIL_LIKELY(current_version.is_free())) {
688 inc_read_lock_count();
689 return read_critical_section{*this, current_version};
690 }
691 // LCOV_EXCL_START
692 if (UNODB_DETAIL_UNLIKELY(current_version.is_obsolete()))
693 return read_critical_section{};
694 UNODB_DETAIL_ASSERT(current_version.is_write_locked());
695 spin_wait_loop_body();
696 // LCOV_EXCL_STOP
697 }
698 }

910 const auto result = load();
911 store(result - 1);
912 return result;
913 }

917 return load() == nullptr;
918 }
919

922 operator T() const noexcept { return load(); }
923

926 return value.load(std::memory_order_relaxed);
927 }
928
930 void store(T new_value) noexcept {
931 value.store(new_value, std::memory_order_relaxed);
932 }

935 in_critical_section(in_critical_section &&) = delete;
936 void operator=(in_critical_section &&) = delete;
937
