627 // Acquire and return a read_critical_section for some lock. This
628 // is done without writing anything on the lock, but it can spin if
629 // the lock is in a transient state (e.g., locked by a writer).
630 //
631 // Note: The returned read_critical_section MAY be marked
632 // [obsolete].
633 //
634 // Note: The caller MUST call read_critical_section::must_restart()
635 // immediately on the result of this method in order to determine if
636 // the node is obsolete.
637 //
638 // @return a read_critical_section which MAY be invalid.
639 [[nodiscard]] read_critical_section try_read_lock() noexcept {
640 while (true) {
641 const auto current_version = version.load_acquire();
642 if (UNODB_DETAIL_LIKELY(current_version.is_free())) {
643 inc_read_lock_count();
644 return read_critical_section{*this, current_version};
645 }
646 // LCOV_EXCL_START
647 if (UNODB_DETAIL_UNLIKELY(current_version.is_obsolete()))
648 return read_critical_section{};
649 UNODB_DETAIL_ASSERT(current_version.is_write_locked());
650 spin_wait_loop_body();
651 // LCOV_EXCL_STOP
652 }
653 }
654
655 // Return a read_critical_section for this optimistic_lock using a
656 // version_tag_type which had been obtained previously. The use case
657 // for this is to fix up the optimistic_lock when a version_tag_type is
658 // read from the stack for an OLC itertor. It bumps the read lock
659 // count to make the code happy but does not do any spin waits or
660 // even look at the current version_tag_type associated with the lock.
661 // When the caller calls read_critical_section::check() on the
662 // returned lock they will figure out whether or not the version is
663 // still valid.
664 [[nodiscard]] read_critical_section rehydrate_read_lock(
665 version_tag_type version_tag) noexcept {
666 // TODO(laurynas) The inc_read_lock_count call should be
667 // refactored to a RCS-creating factory method in optimistic_lock,
668 // removing the need for this comment and cleaning up usage. Not
669 // necessary to do now.
670 inc_read_lock_count();
671 return read_critical_section{*this, version_type(version_tag)};
672 }

777// A gloss for the atomic semantics used to guard loads and stores.
778template <typename T>
779class [[nodiscard]] in_critical_section final {
780 public:
781 constexpr in_critical_section() noexcept = default;
782
783 // cppcheck-suppress noExplicitConstructor
784 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
785 constexpr in_critical_section(T value_) noexcept : value{value_} {}
786
787 in_critical_section(const in_critical_section<T> &) = delete;
788 in_critical_section(in_critical_section<T> &&) = delete;
789
790 ~in_critical_section() noexcept = default;
791
792 in_critical_section<T> &operator=(T new_value) noexcept {
793 store(new_value);
794 return *this;
795 }
796
797 // NOLINTNEXTLINE(cert-oop54-cpp)
798 in_critical_section<T> &operator=(
799 const in_critical_section<T> &new_value) noexcept {
800 store(new_value.load());
801 return *this;
802 }
803
804 void operator=(in_critical_section<T> &&) = delete;
805
806 void operator++() noexcept { store(load() + 1); }
807
808 void operator--() noexcept { store(static_cast<T>(load() - 1)); }
809
810 // NOLINTNEXTLINE(cert-dcl21-cpp)
811 T operator--(int) noexcept {
812 const auto result = load();
813 store(result - 1);
814 return result;
815 }
816
817 template <typename T_ = T,
818 typename = std::enable_if_t<!std::is_integral_v<T_>>>
819 [[nodiscard]] auto operator==(std::nullptr_t) const noexcept {
820 return load() == nullptr;
821 }
822
823 template <typename T_ = T,
824 typename = std::enable_if_t<!std::is_integral_v<T_>>>
825 [[nodiscard]] auto operator!=(std::nullptr_t) const noexcept {
826 return load() != nullptr;
827 }
828
829 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
830 operator T() const noexcept { return load(); }
831
832 [[nodiscard]] T load() const noexcept {
833 return value.load(std::memory_order_relaxed);
834 }
835
836 void store(T new_value) noexcept {
837 value.store(new_value, std::memory_order_relaxed);
838 }
839
840 private:
841 std::atomic<T> value;
842
843 static_assert(std::atomic<T>::is_always_lock_free,
844 "Must use always lock-free atomics");
845}; // class in_critical_section
