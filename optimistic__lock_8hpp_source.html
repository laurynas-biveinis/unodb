387 void write_unlock() noexcept {
388 // This thread has written the previous lock word value, and no other
389 // thread may write it before the unlock, thus we can read it without
390 // ordering.
391 const auto old_lock_word = load_relaxed();
392 UNODB_DETAIL_ASSERT(old_lock_word.is_write_locked());
393
394 const auto new_lock_word = old_lock_word.get() + 2;
395 version.store(new_lock_word, std::memory_order_release);

678 [[nodiscard]] read_critical_section try_read_lock() noexcept {
679 while (true) {
680 const auto current_version = version.load_acquire();
681 if (UNODB_DETAIL_LIKELY(current_version.is_free())) {
682 inc_read_lock_count();
683 return read_critical_section{*this, current_version};
684 }
685 // LCOV_EXCL_START
686 if (UNODB_DETAIL_UNLIKELY(current_version.is_obsolete()))
687 return read_critical_section{};
688 UNODB_DETAIL_ASSERT(current_version.is_write_locked());
689 spin_wait_loop_body();
690 // LCOV_EXCL_STOP
691 }
692 }

751 UNODB_DETAIL_ASSERT(read_lock_count.load(std::memory_order_acquire) > 0);

860class [[nodiscard]] in_critical_section final {
861 public:
863 constexpr in_critical_section() noexcept = default;
864
866 // cppcheck-suppress noExplicitConstructor
867 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
868 constexpr in_critical_section(T value_) noexcept : value{value_} {}
869
871 ~in_critical_section() noexcept = default;
872
874 in_critical_section<T> &operator=(T new_value) noexcept {
875 store(new_value);
876 return *this;
877 }
878
880 // NOLINTNEXTLINE(cert-oop54-cpp)
881 in_critical_section<T> &operator=(
882 const in_critical_section<T> &new_value) noexcept {
883 store(new_value.load());
884 return *this;
885 }
886
888 void operator++() noexcept { store(load() + 1); }
889
891 void operator--() noexcept {
892 // The cast silences MSVC diagnostics about signed/unsigned mismatch.
893 store((static_cast<T>(load() - 1)));
894 }
895
897 // NOLINTNEXTLINE(cert-dcl21-cpp)
898 T operator--(int) noexcept {
899 const auto result = load();
900 store(result - 1);
901 return result;
902 }
903
905 [[nodiscard]] auto operator==(std::nullptr_t) const noexcept {
906 return load() == nullptr;
907 }
908
910 [[nodiscard]] auto operator!=(std::nullptr_t) const noexcept {
911 return load() != nullptr;
912 }
913
915 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
916 operator T() const noexcept { return load(); }
917
919 [[nodiscard]] T load() const noexcept {
920 return value.load(std::memory_order_relaxed);
921 }
922
924 void store(T new_value) noexcept {
925 value.store(new_value, std::memory_order_relaxed);
926 }
927
928 in_critical_section(const in_critical_section<T> &) = delete;
929 in_critical_section(in_critical_section<T> &&) = delete;
930 void operator=(in_critical_section<T> &&) = delete;
931
932 private:
934 std::atomic<T> value;
935
936 static_assert(std::atomic<T>::is_always_lock_free,
937 "Must use always lock-free atomics");
