376 // A read_critical_section (RCS) encapsulates a lock on some node
377 // and the version information that was read for that lock. There
378 // are three different states for an RCS.
379 //
380 // (1) The backing node was obsolete when the RCS was returned by
381 // optimistic_lock::try_read_lock(). This is currently signaled by
382 // [lock==nullptr] internally.
383 //
384 // (2) The RCS was acquired and is valid.
385 //
386 // (3) The RCS has been unlocked and is no longer valid. Note that
387 // in a debug build this also sets [lock=nullptr].
388 class [[nodiscard]] read_critical_section final {
389 public:
390 // construct an RCS for an obsolete node.
391 read_critical_section() noexcept = default;
392
393 read_critical_section(optimistic_lock &lock_,
394 version_type version_) noexcept
395 : lock{&lock_}, version{version_} {}
396
397 read_critical_section &operator=(read_critical_section &&other) noexcept {
398 lock = other.lock;
399 // The current implementation does not need lock == nullptr in the
400 // destructor, thus only reset other.lock in debug builds
401#ifndef NDEBUG
402 other.lock = nullptr;
403#endif
404 version = other.version;
405
406 return *this;
407 }
408
409 // Unlock iff it is not yet unlocked. The read_critical_section
410 // is invalidated by this method and must not be used again by the
411 // caller.
412 //
413 // Note: In a DEBUG build, this clears the [lock] pointer, causing
414 // subsequent use of the RCS to result in a fault, and decrements
415 // the read_lock_count.
416 //
417 // @return true iff the [version] on the optimistic_lock is still
418 // the version that was used to construct this
419 // read_critical_section.
420 [[nodiscard, gnu::flatten]] UNODB_DETAIL_FORCE_INLINE bool try_read_unlock()
421 const noexcept {
422 const auto result = lock->try_read_unlock(version);
423#ifndef NDEBUG
424 lock = nullptr;
425#endif
426 return UNODB_DETAIL_LIKELY(result);
427 }
428
429 // Return true iff the version on the optimistic lock is still the
430 // same version that was used to construct this
431 // read_critical_section (RCS).
432 //
433 // Note: By contract, it is not legal to call this method if the
434 // RCS was marked obsolete when it was constructed. You MUST
435 // detect this situation by calling must_restart() immediately on
436 // obtaining an RCS from optimistic_lock::try_read_lock(). A
437 // failure to do this can lead to the dereference of a nullptr for
438 // the [lock] when you call check().
439 //
440 // Note: By contract, it is not legal to call this method if the
441 // check has already failed. To help catch such situations, in a
442 // DEBUG build, this will clear the [lock] pointer if the check
443 // fails. A subsequent check() call will then dereference a
444 // nullptr and fault the process.
445 //
446 // @return true if the version is unchanged and false if the
447 // caller MUST restart because the version has been changed.
448 [[nodiscard]] bool check() const noexcept {
449 const auto result = lock->check(version);
450#ifndef NDEBUG
451 if (UNODB_DETAIL_UNLIKELY(!result)) lock = nullptr;
452#endif
453 return UNODB_DETAIL_LIKELY(result);
454 }
455
456 // The optimistic_lock::try_read_lock() method MAY return a
457 // read_critical_section (RCS) for an obsolete node. Upon
458 // obtaining the RCS, the caller MUST call this method to
459 // determine whether the node was obsolete and MUST restart if the
460 // method returns false.
461 //
462 // @return false if the node was obsolete at the time that the RCS
463 // was obtained.
464 [[nodiscard]] bool must_restart() const noexcept {
465 return UNODB_DETAIL_UNLIKELY(lock == nullptr);
466 }

473#endif
474 }

555 // Acquire and return a read_critical_section for some lock. This
556 // is done without writing anything on the lock, but it can spin if
557 // the lock is in a transient state (e.g., locked by a writer).
558 //
559 // Note: The returned read_critical_section MAY be marked
560 // [obsolete].
561 //
562 // Note: The caller MUST call read_critical_section::must_restart()
563 // immediately on the result of this method in order to determine if
564 // the node is obsolete.
565 //
566 // @return a read_critical_section which MAY be invalid.
567 [[nodiscard]] read_critical_section try_read_lock() noexcept {
568 while (true) {
569 const auto current_version = version.load();
570 if (UNODB_DETAIL_LIKELY(current_version.is_free())) {
571 inc_read_lock_count();
572 return read_critical_section{*this, current_version};
573 }
574 // LCOV_EXCL_START
575 if (UNODB_DETAIL_UNLIKELY(current_version.is_obsolete()))
576 return read_critical_section{};
577 UNODB_DETAIL_ASSERT(current_version.is_write_locked());
578 spin_wait_loop_body();
579 // LCOV_EXCL_STOP
580 }
581 }

585 // for this is to fix up the optimistic_lock when a version_tag_type is

587 // count to make the code happy but does not do any spin waits or

594 // TODO(laurynas) The inc_read_lock_count call should be

705// A gloss for the atomic semantics used to guard loads and stores.
706template <typename T>
707class [[nodiscard]] in_critical_section final {
708 public:
709 constexpr in_critical_section() noexcept = default;
710
711 // cppcheck-suppress noExplicitConstructor
712 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
713 constexpr in_critical_section(T value_) noexcept : value{value_} {}
714
715 in_critical_section(const in_critical_section<T> &) = delete;
716 in_critical_section(in_critical_section<T> &&) = delete;
717
718 ~in_critical_section() noexcept = default;
719
720 in_critical_section<T> &operator=(T new_value) noexcept {
721 store(new_value);
722 return *this;
723 }
724
725 // NOLINTNEXTLINE(cert-oop54-cpp)
726 in_critical_section<T> &operator=(
727 const in_critical_section<T> &new_value) noexcept {
728 store(new_value.load());
729 return *this;
730 }
731
732 void operator=(in_critical_section<T> &&) = delete;
733
734 void operator++() noexcept { store(load() + 1); }
735
736 void operator--() noexcept { store(static_cast<T>(load() - 1)); }
737
738 // NOLINTNEXTLINE(cert-dcl21-cpp)
739 T operator--(int) noexcept {
740 const auto result = load();
741 store(result - 1);
742 return result;
743 }
744
745 template <typename T_ = T,
746 typename = std::enable_if_t<!std::is_integral_v<T_>>>
747 [[nodiscard]] auto operator==(std::nullptr_t) const noexcept {
748 return load() == nullptr;
749 }
750
751 template <typename T_ = T,
752 typename = std::enable_if_t<!std::is_integral_v<T_>>>
753 [[nodiscard]] auto operator!=(std::nullptr_t) const noexcept {
754 return load() != nullptr;
755 }
756
757 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
758 operator T() const noexcept { return load(); }
759
760 [[nodiscard]] T load() const noexcept {
761 return value.load(std::memory_order_relaxed);
762 }
763
764 void store(T new_value) noexcept {
765 value.store(new_value, std::memory_order_relaxed);
766 }
767
768 private:

772 "Must use always lock-free atomics");
