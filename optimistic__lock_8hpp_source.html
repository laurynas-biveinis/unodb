377 [[nodiscard]] bool cas_acquire(version_type expected,
378 version_type new_val) noexcept {
379 auto expected_val = expected.get();
380 return UNODB_DETAIL_LIKELY(version.compare_exchange_strong(
381 expected_val, new_val.get(), std::memory_order_acquire,
382 std::memory_order_relaxed));

388 void write_unlock() noexcept {
389 // This thread has written the previous lock word value, and no other
390 // thread may write it before the unlock, thus we can read it without
391 // ordering.
392 const auto old_lock_word = load_relaxed();
393 UNODB_DETAIL_ASSERT(old_lock_word.is_write_locked());
394
395 const auto new_lock_word = old_lock_word.get() + 2;
396 version.store(new_lock_word, std::memory_order_release);
397 }

403 void write_unlock_and_obsolete() noexcept {

406 UNODB_DETAIL_ASSERT(!old_lock_word.is_obsolete());
407 UNODB_DETAIL_ASSERT(old_lock_word.is_write_locked());
408#endif
409
410 version.store(version_type::obsolete_lock_word,
411 std::memory_order_release);
412
413 UNODB_DETAIL_ASSERT(load_relaxed().is_obsolete());

683 [[nodiscard]] read_critical_section try_read_lock() noexcept {
684 while (true) {
685 const auto current_version = version.load_acquire();
686 if (UNODB_DETAIL_LIKELY(current_version.is_free())) {
687 inc_read_lock_count();
688 return read_critical_section{*this, current_version};
689 }
690 // LCOV_EXCL_START
691 if (UNODB_DETAIL_UNLIKELY(current_version.is_obsolete()))
692 return read_critical_section{};
693 UNODB_DETAIL_ASSERT(current_version.is_write_locked());
694 spin_wait_loop_body();
695 // LCOV_EXCL_STOP
696 }
697 }

859template <typename T>
860class [[nodiscard]] in_critical_section final {
861 public:
863 constexpr in_critical_section() noexcept = default;
864
866 // cppcheck-suppress noExplicitConstructor
867 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
868 constexpr in_critical_section(T value_) noexcept : value{value_} {}
869
871 ~in_critical_section() noexcept = default;
872
874 // NOLINTNEXTLINE(cert-oop54-cpp)
875 in_critical_section& operator=(
876 const in_critical_section& new_value) noexcept {
877 store(new_value.load());
878 return *this;
879 }
880
882 in_critical_section& operator=(T new_value) noexcept {
883 store(new_value);
884 return *this;
885 }
886
888 void operator++() noexcept { store(load() + 1); }
889
891 void operator--() noexcept {
892 // The cast silences MSVC diagnostics about signed/unsigned mismatch.
893 store((static_cast<T>(load() - 1)));
894 }
895
897 // NOLINTNEXTLINE(cert-dcl21-cpp)
898 T operator--(int) noexcept {
899 const auto result = load();
900 store(result - 1);
901 return result;
902 }
903
905 [[nodiscard]] bool operator==(std::nullptr_t) const noexcept {
906 return load() == nullptr;
907 }
908
910 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
911 operator T() const noexcept { return load(); }
912
914 [[nodiscard]] T load() const noexcept {
915 return value.load(std::memory_order_relaxed);
916 }
917
919 void store(T new_value) noexcept {
920 value.store(new_value, std::memory_order_relaxed);
921 }
922
923 in_critical_section(const in_critical_section&) = delete;
924 in_critical_section(in_critical_section&&) = delete;
925 void operator=(in_critical_section&&) = delete;
926
927 private:
929 std::atomic<T> value;
930
931 static_assert(std::atomic<T>::is_always_lock_free,
932 "Must use always lock-free atomics");
