396 void write_unlock() noexcept {
397 // This thread has written the previous lock word value, and no other
398 // thread may write it before the unlock, thus we can read it without
399 // ordering.
400 const auto old_lock_word = load_relaxed();
401 UNODB_DETAIL_ASSERT(old_lock_word.is_write_locked());
402
403 const auto new_lock_word = old_lock_word.get() + 2;

691 [[nodiscard]] read_critical_section try_read_lock() noexcept {
692 while (true) {
693 const auto current_version = version.load_acquire();
694 if (UNODB_DETAIL_LIKELY(current_version.is_free())) {
695 inc_read_lock_count();
696 return read_critical_section{*this, current_version};
697 }
698 // LCOV_EXCL_START
699 if (UNODB_DETAIL_UNLIKELY(current_version.is_obsolete()))
700 return read_critical_section{};
701 UNODB_DETAIL_ASSERT(current_version.is_write_locked());
702 spin_wait_loop_body();
703 // LCOV_EXCL_STOP
704 }
705 }

763 [[nodiscard]] bool check(version_type locked_version) const noexcept {
764 UNODB_DETAIL_ASSERT(read_lock_count.load(std::memory_order_acquire) > 0);

766#ifndef UNODB_DETAIL_THREAD_SANITIZER

772#endif

944 }
