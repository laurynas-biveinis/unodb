585 // Acquire and return a read_critical_section for some lock. This
586 // is done without writing anything on the lock, but it can spin if
587 // the lock is in a transient state (e.g., locked by a writer).
588 //
589 // Note: The returned read_critical_section MAY be marked
590 // [obsolete].
591 //
592 // Note: The caller MUST call read_critical_section::must_restart()
593 // immediately on the result of this method in order to determine if
594 // the node is obsolete.
595 //
596 // @return a read_critical_section which MAY be invalid.
597 [[nodiscard]] read_critical_section try_read_lock() noexcept {
598 while (true) {
599 const auto current_version = version.load_acquire();
600 if (UNODB_DETAIL_LIKELY(current_version.is_free())) {
601 inc_read_lock_count();
602 return read_critical_section{*this, current_version};
603 }
604 // LCOV_EXCL_START
605 if (UNODB_DETAIL_UNLIKELY(current_version.is_obsolete()))
606 return read_critical_section{};
607 UNODB_DETAIL_ASSERT(current_version.is_write_locked());
608 spin_wait_loop_body();
609 // LCOV_EXCL_STOP
610 }
611 }

613 // Return a read_critical_section for this optimistic_lock using a
614 // version_tag_type which had been obtained previously. The use case
615 // for this is to fix up the optimistic_lock when a version_tag_type is
616 // read from the stack for an OLC itertor. It bumps the read lock
617 // count to make the code happy but does not do any spin waits or
618 // even look at the current version_tag_type associated with the lock.
619 // When the caller calls read_critical_section::check() on the
620 // returned lock they will figure out whether or not the version is
621 // still valid.
622 [[nodiscard]] read_critical_section rehydrate_read_lock(
623 version_tag_type version_tag) noexcept {
624 // TODO(laurynas) The inc_read_lock_count call should be
625 // refactored to a RCS-creating factory method in optimistic_lock,
626 // removing the need for this comment and cleaning up usage. Not
627 // necessary to do now.
628 inc_read_lock_count();
629 return read_critical_section{*this, version_type(version_tag)};
630 }

736template <typename T>
737class [[nodiscard]] in_critical_section final {
738 public:
739 constexpr in_critical_section() noexcept = default;
740
741 // cppcheck-suppress noExplicitConstructor
742 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
743 constexpr in_critical_section(T value_) noexcept : value{value_} {}
744
745 in_critical_section(const in_critical_section<T> &) = delete;
746 in_critical_section(in_critical_section<T> &&) = delete;
747
748 ~in_critical_section() noexcept = default;
749
750 in_critical_section<T> &operator=(T new_value) noexcept {
751 store(new_value);
752 return *this;
753 }
754
755 // NOLINTNEXTLINE(cert-oop54-cpp)
756 in_critical_section<T> &operator=(
757 const in_critical_section<T> &new_value) noexcept {
758 store(new_value.load());
759 return *this;
760 }
761
762 void operator=(in_critical_section<T> &&) = delete;
763
764 void operator++() noexcept { store(load() + 1); }
765
766 void operator--() noexcept { store(static_cast<T>(load() - 1)); }
767
768 // NOLINTNEXTLINE(cert-dcl21-cpp)
769 T operator--(int) noexcept {
770 const auto result = load();
771 store(result - 1);
772 return result;
773 }
774
775 template <typename T_ = T,
776 typename = std::enable_if_t<!std::is_integral_v<T_>>>
777 [[nodiscard]] auto operator==(std::nullptr_t) const noexcept {
778 return load() == nullptr;
779 }
780
781 template <typename T_ = T,
782 typename = std::enable_if_t<!std::is_integral_v<T_>>>
783 [[nodiscard]] auto operator!=(std::nullptr_t) const noexcept {
784 return load() != nullptr;
785 }
786
787 // NOLINTNEXTLINE(google-explicit-constructor,hicpp-explicit-conversions)
788 operator T() const noexcept { return load(); }
789
790 [[nodiscard]] T load() const noexcept {
791 return value.load(std::memory_order_relaxed);
792 }
793
794 void store(T new_value) noexcept {
795 value.store(new_value, std::memory_order_relaxed);
796 }
797
798 private:

802 "Must use always lock-free atomics");
