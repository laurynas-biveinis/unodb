121template <typename To, typename From>
122[[nodiscard, gnu::const]] constexpr To bit_cast(From input) noexcept {
123 // See https://github.com/jfbastien/bit_cast for an implementation using
124 // memcpy (MIT license).
125 static_assert(sizeof(To) == sizeof(From));
126 typename std::aligned_storage<sizeof(To), alignof(To)>::type tmp;
127 std::memcpy(&tmp, &input, sizeof(To));
128 return reinterpret_cast<const To&>(tmp);
129}
