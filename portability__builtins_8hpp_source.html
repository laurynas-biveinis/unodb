24template <typename T>
25[[nodiscard, gnu::const]] constexpr T bswap(T x) noexcept {
26#ifdef UNODB_DETAIL_MSVC
27 static_assert(sizeof(std::uint32_t) ==
28 sizeof(unsigned long)); // NOLINT(runtime/int)
29 static_assert(std::is_same_v<unsigned short, T> || // NOLINT(runtime/int)
30 std::is_same_v<std::uint32_t, T> ||
31 std::is_same_v<std::uint64_t, T>);
32 if constexpr (std::is_same_v<unsigned short, T>) { // NOLINT(runtime/int)
33 return _byteswap_ushort(x);
34 }
35 if constexpr (std::is_same_v<std::uint32_t, T>) {
36 return _byteswap_ulong(x);
37 }
38 if constexpr (std::is_same_v<std::uint64_t, T>) {
39 return _byteswap_uint64(x);
40 }
41#else // UNODB_DETAIL_MSVC
42 static_assert(std::is_same_v<std::uint16_t, T> ||
43 std::is_same_v<std::uint32_t, T> ||
44 std::is_same_v<std::uint64_t, T>);
45 if constexpr (std::is_same_v<std::uint16_t, T>) {
46 return __builtin_bswap16(x);
47 }
48 if constexpr (std::is_same_v<std::uint32_t, T>) {
49 return __builtin_bswap32(x);
50 }
51 if constexpr (std::is_same_v<std::uint64_t, T>) {
52 return __builtin_bswap64(x);
53 }
54#endif // UNODB_DETAIL_MSVC
55}

69template <typename To, typename From>
70[[nodiscard, gnu::const]] constexpr To bit_cast(From input) noexcept {
71 // See https://github.com/jfbastien/bit_cast for an implementation using
72 // memcpy (MIT license).
73 static_assert(sizeof(To) == sizeof(From));
74 typename std::aligned_storage<sizeof(To), alignof(To)>::type tmp;
75 std::memcpy(&tmp, &input, sizeof(To));
76 return reinterpret_cast<const To&>(tmp);
77}
