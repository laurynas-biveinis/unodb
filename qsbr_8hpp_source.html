272 [[nodiscard, gnu::const]] static constexpr type make_from_epoch(

281 [[nodiscard, gnu::const]] static constexpr type inc_thread_count(
282 type word) noexcept {
283 assert_invariants(word);
284
285 const auto result = word + one_thread_in_count;
286
287 assert_invariants(result);
288 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
289 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) ==
290 get_threads_in_previous_epoch(result));
291 UNODB_DETAIL_ASSERT(get_thread_count(word) + 1 == get_thread_count(result));
292
293 return result;
294 }

297 [[nodiscard, gnu::const]] static constexpr type dec_thread_count(
298 type word) noexcept {
299 assert_invariants(word);
300 UNODB_DETAIL_ASSERT(get_thread_count(word) > 0);
301
302 const auto result = word - one_thread_in_count;
303
304 assert_invariants(result);
305 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
306 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) ==
307 get_threads_in_previous_epoch(result));
308 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
309
310 return result;
311 }
312
314 [[nodiscard, gnu::const]] static constexpr type
315 inc_thread_count_and_threads_in_previous_epoch(type word) noexcept {
316 assert_invariants(word);
317
318 const auto result = word + one_thread_and_one_in_previous;
319
320 assert_invariants(result);
321 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
322 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) + 1 ==
323 get_threads_in_previous_epoch(result));
324 UNODB_DETAIL_ASSERT(get_thread_count(word) + 1 == get_thread_count(result));
325
326 return result;
327 }
328
330 [[nodiscard, gnu::const]] static constexpr type
331 dec_thread_count_and_threads_in_previous_epoch(type word) noexcept {
332 assert_invariants(word);
333 UNODB_DETAIL_ASSERT(get_thread_count(word) > 0);
334 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) > 0);
335
336 const auto result = word - one_thread_and_one_in_previous;
337
338 assert_invariants(result);
339 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
340 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) - 1 ==
341 get_threads_in_previous_epoch(result));
342 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
343
344 return result;
345 }

349 [[nodiscard, gnu::const]] static constexpr type inc_epoch_reset_previous(
350 type word) noexcept {
351 assert_invariants(word);
352 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) == 0);
353
354 const auto old_epoch = get_epoch(word);
355 const auto new_epoch_in_word = make_from_epoch(old_epoch.advance());
356 const auto new_thread_count_in_word = word & thread_count_in_word_mask;
357 const auto new_threads_in_previous = (word >> thread_count_in_word_offset) &
358 threads_in_previous_epoch_in_word_mask;
359 const auto result =
360 new_epoch_in_word | new_thread_count_in_word | new_threads_in_previous;
361
362 UNODB_DETAIL_ASSERT(get_epoch(result) == old_epoch.advance());
363 UNODB_DETAIL_ASSERT(get_thread_count(result) == get_thread_count(word));
364 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(result) ==
365 get_thread_count(result));
366 assert_invariants(result);
367
368 return result;
369 }
370
373 [[nodiscard, gnu::const]] static constexpr type
374 inc_epoch_dec_thread_count_reset_previous(type word) noexcept {
375 assert_invariants(word);
376 const auto old_thread_count = get_thread_count(word);
377 UNODB_DETAIL_ASSERT(old_thread_count > 0);
378 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) == 1);
379
380 const auto new_word_with_epoch = make_from_epoch(get_epoch(word).advance());
381 const auto new_thread_count = old_thread_count - 1;
382 const auto new_word_with_thread_count = static_cast<type>(new_thread_count)
383 << thread_count_in_word_offset;
384 const auto new_threads_in_previous = new_thread_count;
385 const auto result = new_word_with_epoch | new_word_with_thread_count |
386 new_threads_in_previous;
387
388 UNODB_DETAIL_ASSERT(get_epoch(word).advance() == get_epoch(result));
389 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
390 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(result) ==
391 get_thread_count(result));
392 assert_invariants(result);
393
394 return result;
395 }

414 static constexpr void assert_invariants(
415 type word UNODB_DETAIL_USED_IN_DEBUG) noexcept {
416#ifndef NDEBUG
417 const auto thread_count = do_get_thread_count(word);
418 UNODB_DETAIL_ASSERT(thread_count <= detail::max_qsbr_threads);
419 const auto threads_in_previous = do_get_threads_in_previous_epoch(word);
420 UNODB_DETAIL_ASSERT(threads_in_previous <= thread_count);
421#endif
422 }
423
425 static constexpr auto thread_count_mask = detail::max_qsbr_threads;
426
427 static_assert((thread_count_mask & (thread_count_mask + 1)) == 0,
428 "Thread count field mask should be 2^n-1");
429
432 static constexpr auto threads_in_previous_epoch_in_word_mask =
433 static_cast<std::uint64_t>(thread_count_mask);
434
436 static constexpr auto thread_count_in_word_offset = 32U;
437
440 static constexpr auto thread_count_in_word_mask =
441 static_cast<std::uint64_t>(thread_count_mask)
442 << thread_count_in_word_offset;
443
445 static constexpr auto epoch_in_word_offset = 62U;
446
449 static constexpr auto one_thread_in_count = 1ULL
450 << thread_count_in_word_offset;
451
455 static constexpr auto one_thread_and_one_in_previous =
456 one_thread_in_count | 1U;

1298inline void qsbr_per_thread::quiescent()
1299#ifndef UNODB_DETAIL_WITH_STATS
1300 noexcept
1301#endif
1302{
1303 UNODB_DETAIL_ASSERT(!paused);
1304 UNODB_DETAIL_ASSERT(active_ptrs.empty());
1305
1306 const auto state = qsbr::instance().get_state();
1307 const auto current_global_epoch = qsbr_state::get_epoch(state);
1308 const auto single_thread_mode = qsbr_state::single_thread_mode(state);
1309
1310 advance_last_seen_epoch(single_thread_mode, current_global_epoch);
1311
1312 if (current_global_epoch != last_seen_quiescent_state_epoch) {
1313 UNODB_DETAIL_ASSERT(current_global_epoch ==
1314 last_seen_quiescent_state_epoch.advance());
1315
1316 last_seen_quiescent_state_epoch = current_global_epoch;
1317#ifdef UNODB_DETAIL_WITH_STATS
1318 qsbr::instance().register_quiescent_states_per_thread_between_epoch_changes(
1319 quiescent_states_since_epoch_change);
1320#endif // UNODB_DETAIL_WITH_STATS
1321 quiescent_states_since_epoch_change = 0;
1322 }
1323
1324 UNODB_DETAIL_ASSERT(current_global_epoch == last_seen_quiescent_state_epoch);
1325 if (quiescent_states_since_epoch_change == 0) {
1326 const auto new_global_epoch =
1327 qsbr::instance().remove_thread_from_previous_epoch(
1328 current_global_epoch
1329#ifndef NDEBUG
1330 ,
1331 last_seen_quiescent_state_epoch
1332#endif
1333 );
1334 UNODB_DETAIL_ASSERT(new_global_epoch == last_seen_quiescent_state_epoch ||
1335 new_global_epoch ==
1336 last_seen_quiescent_state_epoch.advance());
1337
1338 if (new_global_epoch != last_seen_quiescent_state_epoch) {
1339 last_seen_quiescent_state_epoch = new_global_epoch;
1340
1341 UNODB_DETAIL_ASSERT(last_seen_epoch.advance() == new_global_epoch);
1342 update_requests(single_thread_mode, new_global_epoch);
1343
1344#ifdef UNODB_DETAIL_WITH_STATS
1345 qsbr::instance()
1346 .register_quiescent_states_per_thread_between_epoch_changes(1);
1347#endif // UNODB_DETAIL_WITH_STATS
1348 return;
1349 }
1350 }
1351 ++quiescent_states_since_epoch_change;
1352}
