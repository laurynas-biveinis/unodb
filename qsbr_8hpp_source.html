65// Quiescent-state based reclamation (QSBR) memory reclamation scheme. Instead
66// of freeing memory directly, threads register pending deallocation requests to
67// be executed later. Further, each thread notifies when it's not holding any
68// pointers to the shared data structure (is quiescent with respect to that
69// structure). All the threads having passed through a quiescent state
70// constitute a quiescent period, and an epoch change happens at its boundary.
71// At that point all the pending deallocation requests queued before the
72// start of the just-finished quiescent period can be safely executed.
73//
74// For a usage example, see example/example_olc_art.cpp.
75//
76// The implementation borrows some of the basic ideas from
77// https://preshing.com/20160726/using-quiescent-states-to-reclaim-memory/
78
79// If C++ standartisation proposal by A. D. Robison "Policy-based design for
80// safe destruction in concurrent containers" ever gets anywhere, consider
81// changing to its interface, like Stamp-it paper does.
82
83// Two-bit wrapping-around epoch counter. Two epochs can be compared for
84// equality but otherwise are unordered. One bit counter would be enough too,
85// but with two bits we can check more invariants.
86class [[nodiscard]] qsbr_epoch final {
87 public:
88 using epoch_type = std::uint8_t;
89
90 static constexpr epoch_type max = 3U;
91
92 qsbr_epoch(const qsbr_epoch &) noexcept = default;
93 qsbr_epoch(qsbr_epoch &&) noexcept = default;
94 qsbr_epoch &operator=(const qsbr_epoch &) noexcept = default;
95 qsbr_epoch &operator=(qsbr_epoch &&) noexcept = default;
96 ~qsbr_epoch() noexcept = default;
97
98 constexpr explicit qsbr_epoch(epoch_type epoch_val_) : epoch_val{epoch_val_} {
99#ifndef NDEBUG
100 assert_invariant();
101#endif
102 }
103
104 [[nodiscard]] constexpr auto advance(unsigned by = 1) const noexcept {
105#ifndef NDEBUG
106 assert_invariant();
107#endif
108
109 return qsbr_epoch{static_cast<epoch_type>((epoch_val + by) % max_count)};
110 }
111
112 [[nodiscard]] constexpr auto operator==(qsbr_epoch other) const noexcept {
113#ifndef NDEBUG
114 assert_invariant();
115 other.assert_invariant();
116#endif
117
118 return epoch_val == other.epoch_val;
119 }
120
121 [[nodiscard]] constexpr auto operator!=(qsbr_epoch other) const noexcept {
122#ifndef NDEBUG
123 assert_invariant();
124 other.assert_invariant();
125#endif
126
127 return epoch_val != other.epoch_val;
128 }

130 [[gnu::cold]] UNODB_DETAIL_NOINLINE void dump(std::ostream &os) const;
131
132 friend struct qsbr_state;
133
134 qsbr_epoch() noexcept = delete;
135
136 private:
137 static constexpr auto max_count = max + 1U;
138 static_assert((max_count & (max_count - 1U)) == 0);
139
140 epoch_type epoch_val;
141
142#ifndef NDEBUG
143 constexpr void assert_invariant() const noexcept {
144 UNODB_DETAIL_ASSERT(epoch_val <= max);
145 }
146#endif
147};

157// The maximum allowed QSBR-managed thread count is 2^29-1, should be enough for
158// everybody, let's not even bother checking the limit in the Release
159// configuration
160using qsbr_thread_count_type = std::uint32_t;
161
162inline constexpr qsbr_thread_count_type max_qsbr_threads = (2UL << 29U) - 1U;
163
164// Bits are allocated as follows:
165// 0..29: number of threads in the previous epoch
166// 30..31: unused
167// 32..62: total number of threads
168// 63..64: wrapping-around epoch counter
169// Special states: if a thread decrements the number of threads in the previous
170// epoch and observes zero while the total number of threads is greater than
171// zero, then this thread is responsible for the epoch change. The decrement of
172// the last thread in the previous epoch and the epoch bump may happen in a
173// single step, in which case nobody will observe zero threads in the previous
174// epoch.
175struct qsbr_state {
176 using type = std::uint64_t;
177
178 private:
179 [[nodiscard]] static constexpr auto do_get_epoch(type word) noexcept {
180 return qsbr_epoch{
181 static_cast<qsbr_epoch::epoch_type>(word >> epoch_in_word_offset)};
182 }
183
184 [[nodiscard]] static constexpr auto do_get_thread_count(type word) noexcept {
185 const auto result = static_cast<qsbr_thread_count_type>(
186 (word & thread_count_in_word_mask) >> thread_count_in_word_offset);
187 UNODB_DETAIL_ASSERT(result <= max_qsbr_threads);
188 return result;
189 }
190
191 [[nodiscard]] static constexpr auto do_get_threads_in_previous_epoch(
192 type word) noexcept {
193 const auto result = static_cast<qsbr_thread_count_type>(
194 word & threads_in_previous_epoch_in_word_mask);
195 UNODB_DETAIL_ASSERT(result <= max_qsbr_threads);
196 return result;
197 }
198
199 public:
200 [[nodiscard]] static constexpr auto get_epoch(type word) noexcept {
201 assert_invariants(word);
202 return do_get_epoch(word);
203 }
204
205 [[nodiscard]] static constexpr auto get_thread_count(type word) noexcept {
206 assert_invariants(word);
207 return do_get_thread_count(word);
208 }
209
210 [[nodiscard]] static constexpr auto get_threads_in_previous_epoch(
211 type word) noexcept {
212 assert_invariants(word);
213 return do_get_threads_in_previous_epoch(word);
214 }
215
216 [[nodiscard]] static constexpr auto single_thread_mode(type word) noexcept {
217 return get_thread_count(word) < 2;
218 }
219
220 [[gnu::cold]] UNODB_DETAIL_NOINLINE static void dump(std::ostream &os,
221 type word);
222
223 private:
224 friend class qsbr;
225
226 [[nodiscard]] static constexpr auto make_from_epoch(
227 qsbr_epoch epoch) noexcept {
228 const auto result = static_cast<type>(epoch.epoch_val)
229 << epoch_in_word_offset;
230 assert_invariants(result);
231 return result;
232 }
233
234 [[nodiscard]] static constexpr auto inc_thread_count(type word) noexcept {
235 assert_invariants(word);
236
237 const auto result = word + one_thread_in_count;
238
239 assert_invariants(result);
240 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
241 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) ==
242 get_threads_in_previous_epoch(result));
243 UNODB_DETAIL_ASSERT(get_thread_count(word) + 1 == get_thread_count(result));
244
245 return result;
246 }
247
248 [[nodiscard]] static constexpr auto dec_thread_count(type word) noexcept {
249 assert_invariants(word);
250 UNODB_DETAIL_ASSERT(get_thread_count(word) > 0);
251
252 const auto result = word - one_thread_in_count;
253
254 assert_invariants(result);
255 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
256 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) ==
257 get_threads_in_previous_epoch(result));
258 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
259
260 return result;
261 }
262
263 [[nodiscard]] static constexpr auto
264 inc_thread_count_and_threads_in_previous_epoch(type word) noexcept {
265 assert_invariants(word);
266
267 const auto result = word + one_thread_and_one_in_previous;
268
269 assert_invariants(result);
270 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
271 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) + 1 ==
272 get_threads_in_previous_epoch(result));
273 UNODB_DETAIL_ASSERT(get_thread_count(word) + 1 == get_thread_count(result));
274
275 return result;
276 }
277
278 [[nodiscard]] static constexpr auto
279 dec_thread_count_and_threads_in_previous_epoch(type word) noexcept {
280 assert_invariants(word);
281 UNODB_DETAIL_ASSERT(get_thread_count(word) > 0);
282 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) > 0);
283
284 const auto result = word - one_thread_and_one_in_previous;
285
286 assert_invariants(result);
287 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
288 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) - 1 ==
289 get_threads_in_previous_epoch(result));
290 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
291
292 return result;
293 }
294
295 [[nodiscard]] static constexpr auto inc_epoch_reset_previous(
296 type word) noexcept {
297 assert_invariants(word);
298 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) == 0);
299
300 const auto old_epoch = get_epoch(word);
301 const auto new_epoch_in_word = make_from_epoch(old_epoch.advance());
302 const auto new_thread_count_in_word = word & thread_count_in_word_mask;
303 const auto new_threads_in_previous = (word >> thread_count_in_word_offset) &
304 threads_in_previous_epoch_in_word_mask;
305 const auto result =
306 new_epoch_in_word | new_thread_count_in_word | new_threads_in_previous;
307
308 UNODB_DETAIL_ASSERT(get_epoch(result) == old_epoch.advance());
309 UNODB_DETAIL_ASSERT(get_thread_count(result) == get_thread_count(word));
310 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(result) ==
311 get_thread_count(result));
312 assert_invariants(result);
313
314 return result;
315 }
316
317 [[nodiscard]] static constexpr auto inc_epoch_dec_thread_count_reset_previous(
318 type word) noexcept {
319 assert_invariants(word);
320 const auto old_thread_count = get_thread_count(word);
321 UNODB_DETAIL_ASSERT(old_thread_count > 0);
322 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) == 1);
323
324 const auto new_word_with_epoch = make_from_epoch(get_epoch(word).advance());
325 const auto new_thread_count = old_thread_count - 1;
326 const auto new_word_with_thread_count = static_cast<type>(new_thread_count)
327 << thread_count_in_word_offset;
328 const auto new_threads_in_previous = new_thread_count;
329 const auto result = new_word_with_epoch | new_word_with_thread_count |
330 new_threads_in_previous;
331
332 UNODB_DETAIL_ASSERT(get_epoch(word).advance() == get_epoch(result));
333 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
334 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(result) ==
335 get_thread_count(result));
336 assert_invariants(result);
337
338 return result;
339 }
340
341 [[nodiscard]] static constexpr auto
342 dec_thread_count_threads_in_previous_epoch_maybe_advance(
343 type word, bool advance_epoch) noexcept {
344 return UNODB_DETAIL_UNLIKELY(advance_epoch)
345 ? inc_epoch_dec_thread_count_reset_previous(word)
346 : dec_thread_count_and_threads_in_previous_epoch(word);
347 }
348
349 [[nodiscard]] static auto atomic_fetch_dec_threads_in_previous_epoch(
350 std::atomic<type> &word) noexcept;
351
352 static constexpr void assert_invariants(
353 type word UNODB_DETAIL_USED_IN_DEBUG) noexcept {
354#ifndef NDEBUG
355 const auto thread_count = do_get_thread_count(word);
356 UNODB_DETAIL_ASSERT(thread_count <= max_qsbr_threads);
357 const auto threads_in_previous = do_get_threads_in_previous_epoch(word);
358 UNODB_DETAIL_ASSERT(threads_in_previous <= thread_count);
359#endif
360 }
361
362 static constexpr auto thread_count_mask = max_qsbr_threads;
363 static_assert((thread_count_mask & (thread_count_mask + 1)) == 0);
364
365 static constexpr auto threads_in_previous_epoch_in_word_mask =
366 static_cast<std::uint64_t>(thread_count_mask);
367
368 static constexpr auto thread_count_in_word_offset = 32U;
369 static constexpr auto thread_count_in_word_mask =
370 static_cast<std::uint64_t>(thread_count_mask)
371 << thread_count_in_word_offset;
372
373 static constexpr auto epoch_in_word_offset = 62U;
374
375 static constexpr auto one_thread_in_count = 1ULL
376 << thread_count_in_word_offset;
377 static constexpr auto one_thread_and_one_in_previous =
378 one_thread_in_count | 1U;
379};

1054inline void qsbr_per_thread::quiescent()
1055#ifndef UNODB_DETAIL_WITH_STATS
1056 noexcept
1057#endif
1058{
1059 UNODB_DETAIL_ASSERT(!paused);
1060 UNODB_DETAIL_ASSERT(active_ptrs.empty());
1061
1062 const auto state = qsbr::instance().get_state();
1063 const auto current_global_epoch = qsbr_state::get_epoch(state);
1064 const auto single_thread_mode = qsbr_state::single_thread_mode(state);
1065
1066 advance_last_seen_epoch(single_thread_mode, current_global_epoch);
1067
1068 if (current_global_epoch != last_seen_quiescent_state_epoch) {
1069 UNODB_DETAIL_ASSERT(current_global_epoch ==
1070 last_seen_quiescent_state_epoch.advance());
1071
1072 last_seen_quiescent_state_epoch = current_global_epoch;
1073#ifdef UNODB_DETAIL_WITH_STATS
1074 qsbr::instance().register_quiescent_states_per_thread_between_epoch_changes(
1075 quiescent_states_since_epoch_change);
1076#endif // UNODB_DETAIL_WITH_STATS
1077 quiescent_states_since_epoch_change = 0;
1078 }
1079
1080 UNODB_DETAIL_ASSERT(current_global_epoch == last_seen_quiescent_state_epoch);
1081 if (quiescent_states_since_epoch_change == 0) {
1082 const auto new_global_epoch =
1083 qsbr::instance().remove_thread_from_previous_epoch(
1084 current_global_epoch
1085#ifndef NDEBUG
1086 ,
1087 last_seen_quiescent_state_epoch
1088#endif
1089 );
1090 UNODB_DETAIL_ASSERT(new_global_epoch == last_seen_quiescent_state_epoch ||
1091 new_global_epoch ==
1092 last_seen_quiescent_state_epoch.advance());
1093
1094 if (new_global_epoch != last_seen_quiescent_state_epoch) {
1095 last_seen_quiescent_state_epoch = new_global_epoch;
1096
1097 UNODB_DETAIL_ASSERT(last_seen_epoch.advance() == new_global_epoch);
1098 update_requests(single_thread_mode, new_global_epoch);
1099
1100#ifdef UNODB_DETAIL_WITH_STATS
1101 qsbr::instance()
1102 .register_quiescent_states_per_thread_between_epoch_changes(1);
1103#endif // UNODB_DETAIL_WITH_STATS
1104 return;
1105 }
1106 }
1107 ++quiescent_states_since_epoch_change;
1108}
