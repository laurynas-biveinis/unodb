263 [[nodiscard, gnu::const]] static constexpr type make_from_epoch(

272 [[nodiscard, gnu::const]] static constexpr type inc_thread_count(
273 type word) noexcept {
274 assert_invariants(word);
275
276 const auto result = word + one_thread_in_count;
277
278 assert_invariants(result);
279 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
280 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) ==
281 get_threads_in_previous_epoch(result));
282 UNODB_DETAIL_ASSERT(get_thread_count(word) + 1 == get_thread_count(result));
283
284 return result;
285 }

288 [[nodiscard, gnu::const]] static constexpr type dec_thread_count(
289 type word) noexcept {
290 assert_invariants(word);
291 UNODB_DETAIL_ASSERT(get_thread_count(word) > 0);
292
293 const auto result = word - one_thread_in_count;
294
295 assert_invariants(result);
296 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
297 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) ==
298 get_threads_in_previous_epoch(result));
299 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
300
301 return result;
302 }

305 [[nodiscard, gnu::const]] static constexpr type
306 inc_thread_count_and_threads_in_previous_epoch(type word) noexcept {

308

310
311 assert_invariants(result);
312 UNODB_DETAIL_ASSERT(get_epoch(word) == get_epoch(result));
313 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) + 1 ==

321 [[nodiscard, gnu::const]] static constexpr type
322 dec_thread_count_and_threads_in_previous_epoch(type word) noexcept {
323 assert_invariants(word);
324 UNODB_DETAIL_ASSERT(get_thread_count(word) > 0);
325 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) > 0);
326
327 const auto result = word - one_thread_and_one_in_previous;
328
329 assert_invariants(result);

331 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) - 1 ==

333 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
334
335 return result;

340 [[nodiscard, gnu::const]] static constexpr type inc_epoch_reset_previous(
341 type word) noexcept {
342 assert_invariants(word);
343 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) == 0);
344
345 const auto old_epoch = get_epoch(word);
346 const auto new_epoch_in_word = make_from_epoch(old_epoch.advance());
347 const auto new_thread_count_in_word = word & thread_count_in_word_mask;
348 const auto new_threads_in_previous = (word >> thread_count_in_word_offset) &
349 threads_in_previous_epoch_in_word_mask;
350 const auto result =
351 new_epoch_in_word | new_thread_count_in_word | new_threads_in_previous;
352
353 UNODB_DETAIL_ASSERT(get_epoch(result) == old_epoch.advance());
354 UNODB_DETAIL_ASSERT(get_thread_count(result) == get_thread_count(word));
355 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(result) ==
356 get_thread_count(result));
357 assert_invariants(result);
358
359 return result;
360 }
361
364 [[nodiscard, gnu::const]] static constexpr type
365 inc_epoch_dec_thread_count_reset_previous(type word) noexcept {
366 assert_invariants(word);
367 const auto old_thread_count = get_thread_count(word);
368 UNODB_DETAIL_ASSERT(old_thread_count > 0);
369 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(word) == 1);
370
371 const auto new_word_with_epoch = make_from_epoch(get_epoch(word).advance());
372 const auto new_thread_count = old_thread_count - 1;
373 const auto new_word_with_thread_count = static_cast<type>(new_thread_count)
374 << thread_count_in_word_offset;
375 const auto new_threads_in_previous = new_thread_count;
376 const auto result = new_word_with_epoch | new_word_with_thread_count |
377 new_threads_in_previous;
378
379 UNODB_DETAIL_ASSERT(get_epoch(word).advance() == get_epoch(result));
380 UNODB_DETAIL_ASSERT(get_thread_count(word) - 1 == get_thread_count(result));
381 UNODB_DETAIL_ASSERT(get_threads_in_previous_epoch(result) ==
382 get_thread_count(result));
383 assert_invariants(result);
384
385 return result;
386 }

390 [[nodiscard, gnu::const]] static constexpr type

393 return UNODB_DETAIL_UNLIKELY(advance_epoch)

405 static constexpr void assert_invariants(
406 type word UNODB_DETAIL_USED_IN_DEBUG) noexcept {
407#ifndef NDEBUG
408 const auto thread_count = do_get_thread_count(word);
409 UNODB_DETAIL_ASSERT(thread_count <= detail::max_qsbr_threads);
410 const auto threads_in_previous = do_get_threads_in_previous_epoch(word);
411 UNODB_DETAIL_ASSERT(threads_in_previous <= thread_count);
412#endif
413 }
414
416 static constexpr auto thread_count_mask = detail::max_qsbr_threads;
417
418 static_assert((thread_count_mask & (thread_count_mask + 1)) == 0,
419 "Thread count field mask should be 2^n-1");
420
423 static constexpr auto threads_in_previous_epoch_in_word_mask =
424 static_cast<std::uint64_t>(thread_count_mask);
425
427 static constexpr auto thread_count_in_word_offset = 32U;
428
431 static constexpr auto thread_count_in_word_mask =
432 static_cast<std::uint64_t>(thread_count_mask)
433 << thread_count_in_word_offset;
434
436 static constexpr auto epoch_in_word_offset = 62U;
437
440 static constexpr auto one_thread_in_count = 1ULL
441 << thread_count_in_word_offset;
442
446 static constexpr auto one_thread_and_one_in_previous =
447 one_thread_in_count | 1U;
