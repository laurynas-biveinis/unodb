170 static void qsbr_deallocate(void *ptr) {
171#ifdef UNODB_DETAIL_WITH_STATS
172 const auto current_interval_total_dealloc_size_before =
173 unodb::this_thread().get_current_interval_total_dealloc_size();
174#endif // UNODB_DETAIL_WITH_STATS
175 const auto previous_interval_empty_before =
176 unodb::this_thread().previous_interval_requests_empty();
177 const auto current_interval_empty_before =
178 unodb::this_thread().current_interval_requests_empty();
179
180 try {
181 unodb::this_thread().on_next_epoch_deallocate(ptr
182#ifdef UNODB_DETAIL_WITH_STATS
183 ,
184 1
185#endif
186#ifndef NDEBUG
187 ,
188 check_ptr_on_qsbr_dealloc
189#endif
190 );
191 } catch (...) {
192 const auto previous_interval_empty_after =
193 unodb::this_thread().previous_interval_requests_empty();
194 const auto current_interval_empty_after =
195 unodb::this_thread().current_interval_requests_empty();
196#ifdef UNODB_DETAIL_WITH_STATS
197 const auto current_interval_total_dealloc_size_after =
198 unodb::this_thread().get_current_interval_total_dealloc_size();
199 UNODB_EXPECT_EQ(current_interval_total_dealloc_size_before,
200 current_interval_total_dealloc_size_after);
201#endif // UNODB_DETAIL_WITH_STATS
202 UNODB_EXPECT_EQ(previous_interval_empty_before,
203 previous_interval_empty_after);
204 UNODB_EXPECT_EQ(current_interval_empty_before,
205 current_interval_empty_after);
206 throw;
207 }
208
209#ifdef UNODB_DETAIL_WITH_STATS
210 const auto current_interval_total_dealloc_size_after =
211 unodb::this_thread().get_current_interval_total_dealloc_size();
212 const auto current_interval_empty_after =
213 unodb::this_thread().current_interval_requests_empty();
214 const auto single_thread_mode =
215 qsbr_state::single_thread_mode(qsbr::instance().get_state());
216 UNODB_EXPECT_EQ(current_interval_empty_after,
217 (current_interval_total_dealloc_size_after == 0));
218 if (single_thread_mode) {
219 UNODB_EXPECT_TRUE(current_interval_total_dealloc_size_before ==
220 current_interval_total_dealloc_size_after ||
221 current_interval_total_dealloc_size_after == 0);
222 } else {
223 UNODB_EXPECT_GT(current_interval_total_dealloc_size_after, 0);
224 UNODB_EXPECT_TRUE(current_interval_total_dealloc_size_after == 1 ||
225 (current_interval_total_dealloc_size_after ==
226 current_interval_total_dealloc_size_before + 1));
227 }
228#endif // UNODB_DETAIL_WITH_STATS
229 }
