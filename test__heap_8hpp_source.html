64 static void maybe_fail() {
65 // Inspects the fail counter. If non-zero, then bumps the allocation
66 // counter. If that results in the allocation counter reaching or exceeding
67 // the fail counter, then throw std::bad_alloc.
68 if (UNODB_DETAIL_UNLIKELY(paused)) return;
69 const auto fail_counter =
70 fail_on_nth_allocation_.load(std::memory_order_acquire);
71 if (UNODB_DETAIL_UNLIKELY(fail_counter != 0) &&
72 (allocation_counter.fetch_add(1, std::memory_order_relaxed) >=
73 fail_counter - 1)) {
74 throw std::bad_alloc{};
75 }
76 }
